890  {

917  return insert_to_base_node_ref(v);
918  }
919  //v < p

921  {
922  //p == first: insert as first's left child
923  if (p == first_base_node_const_ref())
924  return std::make_pair(insert_as_left_child_of(first_base_node_ref(), v), true);

928  {

1110  void erase(base_node_ref z)
1111  {
1112  base_node_ref y(z->has_missing_child() ? z : z->next_ref());
1113  base_node_ref x;
1114  if (y->has_left())
1115  {
1116  x = y->left_ref;
1117  x->parent_ref = y->parent_ref;
1118  }
1119  else if (y->has_right())
1120  {
1121  x = y->right_ref;
1122  x->parent_ref = y->parent_ref;
1123  }
1124  else x = NULL;
1125  if (is_root(y))
1126  root_ref() = x;
1127  else if (y->is_left_child())
1128  y->parent_ref->left_ref = x;
1129  else y->parent_ref->right_ref = x;
1130 
1131  bool y_was_red(y->is_red);
1132  if (y != z)
1133  {
1134 // Instead of exchanging the keys, exchange the pointers in order to really erase z
1135 // This way iterators to y will not be invalidated, though this operation might be
1136 // more expensive if the data is smaller than de set of pointers to modify
1137 // As well, this allows the keys to be constant
1138 // z->key() = y->key();
1139  y->parent_ref = z->parent_ref;
1140  if (z->is_left_child())
1141  z->parent_ref->left_ref = y;
1142  else z->parent_ref->right_ref = y;
1143  y->left_ref = z->left_ref;
1144  z->left_ref->parent_ref = y;
1145  y->right_ref = z->right_ref;
1146  z->right_ref->parent_ref = y;
1147  y->is_red = z->is_red;
1148  if (is_root(z))
1149  root_ref() = y;
1150  }
1151  if (!y_was_red)
1152  erase_fixup(x);
1153  --d.node_count;
1154  unbalanced_erase(static_cast<node_ref>(z));
1155  }
