551  template<typename CaseNormalizer, typename MetaCodeIterator, typename UCharIterator>
552  void make_union_compressed_delaf_entry_filter(MetaCodeIterator delaf_entry_filter_code_begin, MetaCodeIterator delaf_entry_filter_code_end, compressed_delaf &dico, union_compressed_delaf_entry_filter<UCharIterator> &uudef)
553  {
554  MetaCodeIterator i;
555  while (delaf_entry_filter_code_begin != delaf_entry_filter_code_end)
556  {
557  if (*delaf_entry_filter_code_begin == '+')
558  ++delaf_entry_filter_code_begin;
559  if (delaf_entry_filter_code_begin == delaf_entry_filter_code_end || *delaf_entry_filter_code_begin != '<')
560  fatal_error("Expected a DELAF entry filter code opening '<'\n");
561  ++delaf_entry_filter_code_begin;

563  while (i != delaf_entry_filter_code_end && *i != '>')
564  ++i;

566  fatal_error("Unclosed DELAF entry filter code\n");
567  uudef.compressed_delaf_entry_filters.push_back(make_compressed_delaf_entry_filter<CaseNormalizer, MetaCodeIterator, UCharIterator>(delaf_entry_filter_code_begin, i, dico).release());
568  delaf_entry_filter_code_begin = i;

570  }
