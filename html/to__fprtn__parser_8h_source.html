153  bool operator< (const epsilon_completion &ec) const
154  {
155  if (q_c < ec.q_c)
156  return true;
157  if (ec.q_c < q_c)
158  return false;
159  if (r_c < ec.r_c)
160  return true;
161  if (ec.r_c < r_c)
162  return false;
163  return r_f < ec.r_f;

517  //Compute the r-translations of the input range [input_being, input_end) and add them to the set t
518  void operator()(const in_machine& grammar, bool hasnt_white_at_begin, bool hasnt_white_at_end, out_machine& out)
519  {
520 #ifdef TRACE
521  std::wcout << "Begin parsing\n";
522  //Keep the original input begin for drawing the resulting FPRTN
523 #endif
524  SourceRef input_begin(out.initial_state.second.src_ref);
525  SourceRef input_end(out.global_acceptor_state.second.src_ref);
526 
527  no_op_on_execution_state no_op;
528  acceptor_execution_state_inserter ins_op(out.global_acceptor_state, grammar.initial_state());
529 
530  //Build V_0 as the closure of {(q, {r_I}, lambda, q, 0) | q in Q_I};
531  //If empty input, build initial and final V[0]
532  if (input_begin == input_end)
533  {
534  build_initial_ses(grammar.initial_state(), hasnt_white_at_begin, out, ins_op);
535  //First token is white separated if there are trailing whites at the beginning
536  eclosure(0, input_begin, hasnt_white_at_begin, out, ins_op);
537  }
538  //Else build initial V[0] and the remaining V[i]
539  else
540  {
541  build_initial_ses(grammar.initial_state(), hasnt_white_at_begin, out, no_op);
542  //First token is white separated if there are trailing whites at the beginning
543  eclosure(0, input_begin, hasnt_white_at_begin, out, no_op);
544 
545  std::size_t idx(0);
546  //While the last chart item active set of execution states is not empty and there are input symbols left, compute the next chart item
547  std::size_t last_idx(out.the_chart.size() - 2);
548  SourceRef input_begin_plus_1(input_begin);
549  while (!out.the_chart[idx].other_states.empty() && idx < last_idx)
550  {
551 #ifdef TRACE
552  std::wcout << "----- V[" << static_cast<int>(idx) + 1 << "] -----" << std::endl;
553 #endif
554  ++input_begin_plus_1;
555  translate_symbol(idx, input_begin, input_begin_plus_1, out, no_op);
556  ++idx;
557  eclosure(idx, input_begin_plus_1, (*input_begin)->end == (*input_begin_plus_1)->begin, out, no_op);
558  ++input_begin;
559  }
560 
561  //Compute last chart item
562  if (!out.the_chart[idx].other_states.empty())
563  {
564 #ifdef TRACE
565  std::wcout << "----- V[" << static_cast<int>(idx) + 1 << "] -----" << std::endl;
566 #endif
567  ++input_begin_plus_1;
568  translate_symbol(idx, input_begin, input_begin_plus_1, out, ins_op);
569  ++idx;
570  //Next token is the input end... it is white separated if there are trailing whites at the end
571  eclosure(idx, input_begin_plus_1, hasnt_white_at_end, out, ins_op);
572  ++input_begin;
573  }
574  }
575 #ifdef TRACE
576  std::wcout << "***** Paused ESs *****\n";
577  std::wcout << "----- V[" << static_cast<int>(0) << "] -----\n";
578  paused_execution_state_map_const_iterator x_p_it(out.the_chart[0].pess.begin());
579  for (paused_execution_state_map_const_iterator x_p_it(out.the_chart[0].pess.begin()); x_p_it != out.the_chart[0].pess.end(); ++x_p_it)
580  static_cast<const paused_execution_state&>(*x_p_it).serialize(std::wcout) << std::endl;
581  input_begin = out.initial_state.second.src_ref;
582  std::size_t idx(1);
583  for (; input_begin != input_end; ++input_begin, ++idx)
584  {
585  std::wcout << "----- V[" << static_cast<int>(idx) << "] -----\n";
586  for (x_p_it = out.the_chart[idx].pess.begin(); x_p_it != out.the_chart[idx].pess.end(); ++x_p_it)
587  static_cast<const paused_execution_state&>(*x_p_it).serialize(std::wcout) << std::endl;
588  }
589  std::wcout << "***** End of paused ESs *****\n";
590  std::wcout << "End parsing\n";
591  write_fprtn_dot("dirty_parses", "../dirty_parses.dot", out);
592 #endif
593  }
594  };
