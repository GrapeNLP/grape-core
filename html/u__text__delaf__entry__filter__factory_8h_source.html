544  void make_u_union_text_delaf_entry_filter(MetaCodeIterator delaf_entry_filter_code_begin, MetaCodeIterator delaf_entry_filter_code_end, u_text_delaf<CaseNormalizer> &dico, u_union_text_delaf_entry_filter<CaseNormalizer> &uudef)
545  {

547  while (delaf_entry_filter_code_begin != delaf_entry_filter_code_end)
548  {
549  if (*delaf_entry_filter_code_begin == '+')
550  ++delaf_entry_filter_code_begin;
551  if (delaf_entry_filter_code_begin == delaf_entry_filter_code_end || *delaf_entry_filter_code_begin != '<')
552  fatal_error("Expected a DELAF entry filter code opening '<'\n");
553  ++delaf_entry_filter_code_begin;
554  i = delaf_entry_filter_code_begin;
555  while (i != delaf_entry_filter_code_end && *i != '>')
556  ++i;

558  fatal_error("Unclosed DELAF entry filter code\n");
559  uudef.delaf_entry_filters.push_back(make_u_text_delaf_entry_filter(delaf_entry_filter_code_begin, i, dico).release());
560  delaf_entry_filter_code_begin = i;
561  ++delaf_entry_filter_code_begin;

565  template<typename CaseNormalizer, typename MetaCodeIterator>
566  void make_u_semantic_traits_to_edit(MetaCodeIterator delaf_entry_filter_code_begin, MetaCodeIterator delaf_entry_filter_code_end, u_text_delaf<CaseNormalizer> &dico, typename u_text_delaf<CaseNormalizer>::ust_set &semantic_traits_to_add, typename u_text_delaf<CaseNormalizer>::ust_set &semantic_traits_to_substract)
567  {
568  MetaCodeIterator current_char(delaf_entry_filter_code_begin);
569  if (current_char == delaf_entry_filter_code_end || *current_char == flexional_traits_separator)
570  fatal_error("DELAF entry filter having an explicit empty mandatory POS and/or forbidden set of PPOOSS and/or set of semantic constraints\n");
571  //First semantic constrait is mandatory by default (if '+' is omitted)
572  bool mandatory(true);
573  ust_string::ref semantic_trait_ref;
574  //While there are semantic traits left, get next
575  while (current_char != delaf_entry_filter_code_end && *current_char != flexional_traits_separator)
576  {
577  if (*current_char == mandatory_semantic_trait_mark || *current_char == forbidden_semantic_trait_mark)
578  {
579  mandatory = *current_char == mandatory_semantic_trait_mark;
580  ++current_char;
581  }
582  delaf_entry_filter_code_begin = current_char;
583  if (current_char == delaf_entry_filter_code_end || *current_char == mandatory_semantic_trait_mark || *current_char == forbidden_semantic_trait_mark || *current_char == flexional_traits_separator)
584  fatal_error("Explicit empty semantic trait\n");
585  do ++current_char;
586  while (current_char != delaf_entry_filter_code_end && *current_char != mandatory_semantic_trait_mark && *current_char != forbidden_semantic_trait_mark && *current_char != flexional_traits_separator);
587  semantic_trait_ref = dico.semantic_traits.get(delaf_entry_filter_code_begin, current_char);
588  if (mandatory)
589  {
590  //If semantic trait not already used in the dictionary, add it to the set of semantic traits
591  if (!semantic_trait_ref)
592  semantic_trait_ref = &dico.semantic_traits.epsilon().concat(delaf_entry_filter_code_begin, current_char);
593  semantic_traits_to_add.insert(semantic_trait_ref);
594  }
595  //If semantic trait present in the dictionary, add it to the set of semantic traits to substract
596  //Otherwise there is no need to substract it
597  else if (semantic_trait_ref)
598  semantic_traits_to_substract.insert(semantic_trait_ref);
599  }
600  }
