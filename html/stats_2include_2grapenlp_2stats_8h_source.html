176  template<typename CharT, typename Traits>
177  std::basic_istream<CharT, Traits>& operator>> (std::basic_istream<CharT, Traits> &in, stats &s)
178  {
179  s.reset();
180  //Skip up to next parser header box
181  skip_up_to_char(in, '*');
182  skip_up_to_char(in, ' ');
183  std::string parser_name;
184  std::getline(in, parser_name, '*');
185  std::size_t i(0);
186  std::size_t j(parser_name.find(' '));
187  if (!j)
188  fatal_error("Empty parser name\n");
189  if (j == std::string::npos)
190  fatal_error("Invalid input file format: missing space after algorithm name\n");
191  s.parser_name = parser_name.substr(0, j);
192  i = j + 1;
193  while (i < parser_name.size() - 1)
194  {
195  j = parser_name.find(' ', i);
196  if (j == std::string::npos)
197  fatal_error("Invalid input file format: missing space after algorithm parameter\n");
198  if (i == j)
199  fatal_error("Empty parser parameter\n");
200  if ((j - i) < 2)
201  fatal_error("Parser parameter has less than 2 chars\n");
202  if (!(s.pitvm.insert(std::make_pair(parser_name[i+1], parser_name.substr(i, j - i))).second))
203  fatal_error("Repeated param id '%c'\n", parser_name[i+1]);
204  i = j + 1;
205  }
206  skip_line(in);
207  skip_line(in);
208  set_trie<char>::string::ref stsr;
209  std::string line;
210  bool asps_sd_cv_p_min_max_read(false);
211  while(!in.eof())
212  {
213  std::getline(in, line);
214  stsr = the_stats_label_set_trie.get_longest_match(line.begin(), line.end());
215  if (!stsr->data)
216  continue;
217  if (stsr == the_stats_label_dico.parser_header) return in;
218  else if (stsr == the_stats_label_dico.grammar_states) s.grammar_states = get_ending_figure_string(line).first;
219  else if (stsr == the_stats_label_dico.grammar_transitions) s.grammar_transitions = get_ending_figure_string(line).first;
220  else if (stsr == the_stats_label_dico.input_chars) s.input_chars = get_ending_figure_string(line).first;
221  else if (stsr == the_stats_label_dico.input_tokens) s.input_tokens = get_ending_figure_string(line).first;
222  else if (stsr == the_stats_label_dico.input_sentences) s.input_sentences = get_ending_figure_string(line).first;
223  else if (stsr == the_stats_label_dico.input_parses) s.input_parses = get_ending_figure_string(line).first;
224  else if (stsr == the_stats_label_dico.average_chars_per_token) s.average_chars_per_token = get_ending_figure_string(line).first;
225  else if (stsr == the_stats_label_dico.average_tokens_per_sentence) s.average_tokens_per_sentence = get_ending_figure_string(line).first;
226  else if (stsr == the_stats_label_dico.average_chars_per_sentence) s.average_chars_per_sentence = get_ending_figure_string(line).first;
227  else if (stsr == the_stats_label_dico.average_parses_per_sentence) s.average_parses_per_sentence = get_ending_figure_string(line).first;
228  else if (stsr == the_stats_label_dico.total_chars) s.total_chars = get_ending_figure_string(line).first;
229  else if (stsr == the_stats_label_dico.total_tokens) s.total_tokens = get_ending_figure_string(line).first;
230  else if (stsr == the_stats_label_dico.total_sentences) s.total_sentences = get_ending_figure_string(line).first;
231  else if (stsr == the_stats_label_dico.total_parses) s.total_parses = get_ending_figure_string(line).first;
232  else if (stsr == the_stats_label_dico.total_elapsed_seconds) s.total_elapsed_seconds = get_ending_figure_string(line).first;
233  else if (stsr == the_stats_label_dico.average_seconds_per_sentence) s.average_seconds_per_sentence = get_ending_figure_string(line).first;
234  else if (stsr == the_stats_label_dico.asps_sd_cv_p_min_max)
235  {
236  if (asps_sd_cv_p_min_max_read)
237  get_sd_cv_p_min_max(line, s.iasps_sd, s.iasps_cv, s.iasps_p, s.iasps_min, s.iasps_max);
238  else
239  {
240  get_sd_cv_p_min_max(line, s.asps_sd, s.asps_cv, s.asps_p, s.asps_min, s.asps_max);
241  asps_sd_cv_p_min_max_read = true;
242  }
243  }
244  else if (stsr == the_stats_label_dico.average_sentences_per_second) s.average_sentences_per_second = get_ending_figure_string(line.substr(0, line.size() - 4)).first;
245  }
246  return in;
247  }
