621  * Given a set of states of a FPRTN, removes the ones that are marked as unreachable
622  */
623  void remove_unreachable_states(state_ref_list &states)
624  {
625  state_ref_list_iterator n(states.begin());
626  state_ref_list_iterator n_to_erase;
627  while (n != states.end())
628  {
629  //If the state is not reachable,
630  if (!(*n)->reached_flag)
631  {

633  std::wcout << L"removing state " << (*n)->wlabel() << std::endl;
634 #endif

636  (*n)->remove_foreign_transitions();
637  //advance initial state iterator before deleting the current initial state!!
638  n_to_erase = n;
639  ++n;

641  states.erase(n_to_erase);
642  }

644  else
645  {
646 #ifdef TRACE
647  std::wcout << (*n)->wlabel() << L" is useful" << std::endl;

667  //Remove all paths which do not reach the specified state from anywhere.
668  //Notice that after parsing, every state in the FPRTN is derivable from an initial state except
669  //the ones belonging to uncompleted calls, and that the algorithm creates a single
670  //global final state, thus by specifying this final state as argument for this method
671  //the FPRTN will be pruned (every useless path will be deleted)
672  fprtn& output_fprtn_prune(state_ref path_target_state)
673  {
674 #ifdef TRACE
675  std::wcout << L"Begin removing unused paths\n";
676 #endif
677 
678  //If there are no popping transitions towards the final state,
679  //the FPRTN is equivalent to an empty FPRTN, so clear it and return
680  if (path_target_state->incoming_filtered_pop_transitions.empty())
681  {
682 #ifdef TRACE
683  std::wcout << L"FPRTN cleared: no interpretations found\n";
684 #endif
685  clear();
686  return *this;
687  }

689  state_ref_queue pending_states;
690  state_ref s;
691 /*
692  typename fprtn<SourceRef, TagInput>::outgoing_epsilon_transition_iterator oet_it;
693  typename fprtn<SourceRef, TagInput>::outgoing_call_transition_iterator ocallt_it;
694  typename fprtn<SourceRef, TagInput>::outgoing_filtered_pop_transition_iterator ofpt_it;
695 */
696  //Otherwise, begin by marking the final state as a reachable state
697  path_target_state->reached_flag = true;
698 #ifdef TRACE
699  std::wcout << L"start search of reversely reachable states from " << path_target_state->wlabel() << std::endl;
700 #endif
701  //And add the final state to the queue of states to process
702  pending_states.push(path_target_state);
703 
704  //While there are states to pending to be processed
705  while (!pending_states.empty())
706  {
707  //Pop next state to process
708  s = pending_states.front();
709  pending_states.pop();
710  //For every consuming transition coming to the current state,
711  //mark its source state as reachable and add it to the queue
712  for (outgoing_consuming_transition_iterator oconst_it(s->incoming_consuming_transitions.begin()); oconst_it != s->incoming_consuming_transitions.end(); ++oconst_it)
713  {
714  if (!oconst_it->target->reached_flag)
715  {
716 #ifdef TRACE
717  std::wcout << L"reached state " << oconst_it->target->wlabel() << L" from " << s->wlabel() << std::endl;
718 #endif
719  oconst_it->target->reached_flag = true;
720  pending_states.push(oconst_it->target);
721  }
722  }
723  //For every consuming transition coming to the current state,
724  //mark its source state as reachable and add it to the queue
725  for (outgoing_epsilon_transition_iterator oet_it(s->incoming_epsilon_transitions.begin()); oet_it != s->incoming_epsilon_transitions.end(); ++oet_it)
726  {
727  if (!oet_it->target->reached_flag)
728  {
729 #ifdef TRACE
730  std::wcout << L"reached state " << oet_it->target->wlabel() << L" from " << s->wlabel() << std::endl;
731 #endif
732  oet_it->target->reached_flag = true;
733  pending_states.push(oet_it->target);
734  }
735  }
736  //For every call transition coming to this state,
737  //mark its source state as reachable and add it to the queue
738  for (outgoing_call_transition_iterator ocallt_it(s->incoming_call_transitions.begin()); ocallt_it != s->incoming_call_transitions.end(); ++ocallt_it)
739  {
740  if (!ocallt_it->target->reached_flag)
741  {
742 #ifdef TRACE
743 
744  std::wcout << L"reached state " << ocallt_it->target->wlabel() << L" from " << s->wlabel() << std::endl;
745 #endif
746  ocallt_it->target->reached_flag = true;
747  pending_states.push(ocallt_it->target);
748  }
749  }
750  //For every popping transition coming to this state,
751  //mark its source state as reachable and add it to the queue
752  for (incoming_filtered_pop_transition_iterator ifpt_it(s->incoming_filtered_pop_transitions.begin()); ifpt_it != s->incoming_filtered_pop_transitions.end(); ++ifpt_it)
753  {
754  if (!ifpt_it->target->reached_flag)
755  {
756 #ifdef TRACE
757 
758  std::wcout << L"reached state " << ifpt_it->target->wlabel() << L" from " << s->wlabel() << std::endl;
759 #endif
760  ifpt_it->target->reached_flag = true;
761  pending_states.push(ifpt_it->target);
762  }
763  }
764  }
765 
766  remove_unreachable_states();
767 
768 #ifdef TRACE
769  std::wcout << L"End removing unused paths\n";
770 #endif
771  return *this;
772  }
