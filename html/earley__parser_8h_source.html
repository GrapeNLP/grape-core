303  //Epsilon-completion map: maps deletable states of the grammar (states that when called, can lead to a call completion without input consumption)
304  //with pairs (r_c, set of r_f's), where r_c is the output state to be called (corresponding to the deletable called state)
305  //and the r_f's are the final output states from where the call is epsilon-completed
306  //Given an output state r_s from where the epsilon-completed call is performed and an output target state r_t,
307  // we complete the call by adding the call transition r_s --r_c--> r_t and the filtered pop transitions
308  //r_f -> r_t, for every r_f in the set of r_f's
309  struct epsilon_completion
310  {
311  state_const_ref q;
312  const blackboard *b_ref;
313 
314  //Constructor for q lower bound
315  epsilon_completion(state_const_ref q_): q(q_), b_ref(NULL)
316  {}
317 
318  epsilon_completion(state_const_ref q_, const blackboard *b_ref_): q(q_), b_ref(b_ref_)
319  {}
320 
321  bool operator< (const epsilon_completion &ec) const
322  {
323  if (q < ec.q)
324  return true;
325  if (ec.q < q)
326  return false;
327  //Dummy epsilon completion is the only one having a NULL blackboard pointer

329  }

331  std::ptrdiff_t compare_3w(const epsilon_completion &ec) const
332  {
333  std::ptrdiff_t result(reinterpret_cast<std::ptrdiff_t>(q) - reinterpret_cast<std::ptrdiff_t>(ec.q));
334  if (result)
335  return result;
336  //Dummy epsilon completion is the only one having a NULL blackboard pointer
337  if ((result = static_cast<std::ptrdiff_t>(!ec.b_ref) - static_cast<std::ptrdiff_t>(!b_ref)))
338  return result;
339  if ((result = static_cast<std::ptrdiff_t>(!ec.b_ref) - static_cast<std::ptrdiff_t>(!b_ref)))
340  return result;
341  return grapenlp::compare_3w<blackboard>()(*b_ref, *ec.b_ref);
342  }

494  }

551  //Earley scanner: adds a new item to the parsing chart and initializes it with the attainable set of execution states from the former last chart item set of execution states through the consumption of an input symbol sigma
552  template<typename ExtraInsertOp>
553  void translate_symbol(size_t idx, SourceRef in_i, ExtraInsertOp op)
554  {
555  chart_item &v = the_chart[idx];
556  chart_item &w = the_chart[idx + 1];
557  //For each (q, b, lambda, q_h, i) in v
558  for (active_execution_state_set_const_iterator x_s_it(v.aess.begin()); x_s_it != v.aess.end(); ++x_s_it)
559  {
560  //For each (q', lambda) = delta(q, (sigma, epsilon))
561  //Process deleting transitions
562  for (outgoing_deleting_transition_set_const_iterator odt_it(x_s_it->q->outgoing_deleting_transitions.begin()); odt_it != x_s_it->q->outgoing_deleting_transitions.end(); ++odt_it)
563  {
564  if (input_match(*in_i, odt_it->input))
565  {
566  //w = w union (q', b, lambda, q_h, i)
567  std::pair<active_execution_state_set_const_iterator, bool> result(w.aess.insert(active_execution_state(static_cast<state_const_ref>(odt_it->target), x_s_it->b, x_s_it->q_h, x_s_it->i)));
568  if (result.second)
569  {
570 #ifdef TRACE
571  result.first->serialize(std::wcout) << L" (";
572  input_serializer()(std::wcout, *in_i) << L" : <E>)" << std::endl;
573 #endif
574  e.push(&(*result.first));
575  op(*result.first);
576  }
577  }
578  }
579  //For each (q', lambda) = delta(q, (sigma, g))
580  //Process translating transitions
581  for (outgoing_translating_transition_set_const_iterator ott_it(x_s_it->q->outgoing_translating_transitions.begin()); ott_it != x_s_it->q->outgoing_translating_transitions.end(); ++ott_it)
582  {
583  if (input_match(*in_i, ott_it->input))
584  {
585  //w = w union (q', bg, lambda, q_h, i)
586  blackboard b_t(x_s_it->b);
587  std::pair<active_execution_state_set_const_iterator, bool> result(w.aess.insert(active_execution_state(static_cast<state_const_ref>(ott_it->target), gamma(b_t, in_i, ott_it->output), x_s_it->q_h, x_s_it->i)));
588  if (result.second)
589  {
590 #ifdef TRACE
591  result.first->serialize(std::wcout) << L" (";
592  input_serializer()(std::wcout, *in_i) << L" : ";
593  tag_output_serializer()(std::wcout, ott_it->output) << L')' << std::endl;
594 #endif
595  e.push(&(*result.first));
596  op(*result.first);
597  }
598  }
599  }
600  }

618  //Compute the r-translations of the input range [input_being, input_end) and add them to the set t
619  blackboard_set& operator()(const machine& grammar, SourceRef input_begin, SourceRef input_end, bool hasnt_white_at_begin, bool hasnt_white_at_end, blackboard_set& t, const blackboard &empty_blackboard = blackboard())
620  {
621 #ifdef TRACE
622  std::wcout << L"Begin parsing\n";
623 #endif
624  //Initialize chart
625  std::size_t chart_size(std::distance(input_begin, input_end) + 1);
626  the_chart.reset(chart_size);
627 
628  no_op_on_execution_state no_op;
629  acceptor_execution_state_inserter ins_op(t, grammar.initial_state());
630 
631  //Build V_0 as the closure of {(q, epsilon, lambda, q, 0) | q in Q_I};
632  //If empty input, build initial and final V[0]
633  if (input_begin == input_end)
634  {
635  build_initial_ses(grammar.initial_state(), hasnt_white_at_begin, ins_op, empty_blackboard);
636  //First token is white separated if there are trailing whites at the beginning
637  eclosure(0, input_begin, hasnt_white_at_begin, ins_op, empty_blackboard);
638  }
639  //Else build initial V[0] and the remaining V[i]
640  else
641  {
642  build_initial_ses(grammar.initial_state(), hasnt_white_at_begin, no_op, empty_blackboard);
643  //First token is white separated if there are trailing whites at the beginning
644  eclosure(0, input_begin, hasnt_white_at_begin, no_op, empty_blackboard);
645 
646  std::size_t idx(0);
647  //While the last chart item active set of execution states is not empty and there are input symbols left, compute the next chart item
648  std::size_t last_idx(chart_size - 2);
649  SourceRef input_begin_plus_1(input_begin);
650  while (!the_chart[idx].aess.empty() && idx < last_idx)
651  {
652 #ifdef TRACE
653  std::wcout << "----- V[" << static_cast<int>(idx) + 1 << "] -----" << std::endl;
654 #endif
655  ++input_begin_plus_1;
656  translate_symbol(idx, input_begin, no_op);
657  ++idx;
658  eclosure(idx, input_begin_plus_1, (*input_begin)->end == (*input_begin_plus_1)->begin, no_op, empty_blackboard);
659  ++input_begin;
660  }
661 
662  //Compute last chart item
663  if (!the_chart[idx].aess.empty())
664  {
665 #ifdef TRACE
666  std::wcout << "----- V[" << static_cast<int>(idx) + 1 << "] -----" << std::endl;
667 #endif
668  ++input_begin_plus_1;
669  translate_symbol(idx, input_begin, ins_op);
670  ++idx;
671  //Next token is the input end... it is white separated if there are trailing whites at the end
672  eclosure(idx, input_begin_plus_1, hasnt_white_at_end, ins_op, empty_blackboard);
673  ++input_begin;
674  }
675  }
676  the_chart.reset();
677 #ifdef TRACE
678  std::wcout << L"End parsing\n";
679 #endif
680  return t;
681  }
