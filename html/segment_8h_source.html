167  segment_map& insert(SourceRef src_ref, SourceRef begin_src_ref, SourceRef end_src_ref, const out_bound<Id> &ob)
168  {
169 //#ifdef TRACE
170 // std::wcout << L"Inserting out_bound " << ob << std::endl;
171 //#endif
172  iterator i(this->lower_bound(ob.name));
173  //If there is no other bound with the same name
174  if (i == this->end() || i->first != ob.name)
175  {
176  //Insert left-open/right-closed segment
177  if (!ob.left)
178 #ifdef TRACE
179  base_type::insert(i, value_type(ob.name, segment<SourceRef>(begin_src_ref, begin_src_ref, false, src_ref, true)));
180 #else
181  base_type::insert(i, value_type(ob.name, segment<SourceRef>(begin_src_ref, false, src_ref, true)));
182 #endif
183  //Insert closed empty segment
184  else if (ob.right)
185 #ifdef TRACE
186  base_type::insert(i, value_type(ob.name, segment<SourceRef>(begin_src_ref, src_ref, true, src_ref, true)));
187 #else
188  base_type::insert(i, value_type(ob.name, segment<SourceRef>(src_ref, true, src_ref, true)));
189 #endif
190  //Insert left-closed/right-open segment
191 #ifdef TRACE
192  else base_type::insert(i, value_type(ob.name, segment<SourceRef>(begin_src_ref, src_ref, true, end_src_ref, false)));
193 #else
194  else base_type::insert(i, value_type(ob.name, segment<SourceRef>(src_ref, true, end_src_ref, false)));
195 #endif
196  return *this;
197  }
198  //If segment's right bound is undefined and new bound is a right bound, close segment's right bound
199  if (!i->second.is_end_defined() && !ob.left)
200  i->second.set_end(src_ref);
201  return *this;
202  }

241  //This is for the composer; no need for an unweighted converse version, for now
242  segment_map& insert(const segment_map &sm)
243  {
244  if (sm.empty())
245  return *this;
246  //Set sm's position to the first segment
247  const_iterator j(sm.begin());
248  //Search insertion point in *this for first sm's segment
249  iterator i = base_type::lower_bound(j->first);
250 
251  //If insertion point is not yet at the end
252  while (i != this->end())
253  {
254  //If sm's segment is not present in *this
255  if (i->first != j->first)
256  base_type::insert(i, value_type(j->first, segment<SourceRef>(j->second)));
257  //Else if *this segment is right open and sm segment is right closed, close also *this segment's right bound
258  else if (!i->second.is_end_defined() && !j->second.is_begin_defined())
259  i->second.set_end(j->second.get_end());
260  //Advance to next sm's segment
261  ++j;
262  //If there are no more segments to insert, return
263  if (j == sm.end())
264  return *this;
265  //Else search for new corresponding insertion point in *this
266  i = base_type::lower_bound(j->first);
267  }
268  //Insert remaining sm segments at the end of *this
269  do
270  { base_type::insert(this->end(), value_type(j->first, segment<SourceRef>(j->second))); }
271  while (++j != sm.end());
272  return *this;
273  }

275  bool operator< (const segment_map &sm) const
276  {
277  const_iterator i1(this->begin());
278  const_iterator i2(sm.begin());
279 
280  while (true)
281  {
282  if (i1 == this->end())
283  {
284  //Equal
285  if (i2 == sm.end())
286  return false;
287  //Less: this segment_map has less segments than sm
288  return true;
289  }
290  if (i2 == sm.end())
291  //Greater: this segment_map has more segments than sm
292  return false;
293  //Current segment name of this segment_map is different than the one of sm: return this_segment_name less_than sm_segment_name
294  if (i1->first < i2->first)
295  return true;
296  if (i2->first < i1->first)
297  return false;
298  if (i1->second < i2->second)
299  //Less: current segment bounds of this segment_map are less than the ones of sm
300  return true;
301  if (i2->second < i1->second)
302  //Greater: current segment bounds of this segment_map are greater than the ones of sm
303  return false;
304  //Advance to next pair of segments
305  ++i1;
306  ++i2;
307  }
