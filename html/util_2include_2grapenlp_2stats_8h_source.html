89  void compute_stats()
90  {
91  mean_ /= static_cast<stat_type>(samples.size());
92  stat_type diff;
93  stat_type sum_diff_sq(0.0L);
94  sample_array_iterator i(samples.begin());
95  for (; i != samples.end(); ++i)
96  {
97  diff = mean_ - static_cast<stat_type>(*i);
98  sum_diff_sq += diff * diff;
99  }
100  std_dev_ = std::sqrt(sum_diff_sq / (samples.size() - 1));
101  cv_ = (static_cast<stat_type>(100.0)) * (std_dev_ / mean_);
102  up_ = mean_ + std_dev_;
103  down_ = mean_ - std_dev_;
104  std::size_t in_count(0);
105  for (i = samples.begin(); i != samples.end(); ++i)
106  {
107  if (down_ <= static_cast<stat_type>(*i) && static_cast<stat_type>(*i) <= up_)
108  ++in_count;
109  }
110  p_ = (static_cast<stat_type>(100.0)) * (static_cast<stat_type>(in_count) / static_cast<stat_type>(samples.size()));
