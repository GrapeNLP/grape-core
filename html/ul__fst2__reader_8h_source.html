330  int u_read_int_or_eol(FILE *f, bool &eol, unsigned int &line_count)
331  {
332  unichar c;
333  //Skip spaces
334  do c=(unichar)u_fgetc(f); while (c==' ');
335  //Next char must be a digit, a minus or an EOL
336  if (!(u_is_digit(c) || c == '-' || u_is_eol(c)))
337  fatal_error("In line %u of .fst2 file: unexpected char '%c' (expected digit)\n", line_count, c);
338 
339  //EOL
340  if (u_is_eol(c))
341  {
342  eol = true;
343  return 0;
344  }
345 
346  eol = false;
347  int sign;
348 
349  //Minus
350  if (c=='-')
351  {
352  sign = -1;
353  c=(unichar)u_fgetc(f);
354  if (!u_is_digit(c))
355  fatal_error("Unexpected character in fst2 (read int 2): '%c'\n", c);
356  }
357  else sign = 1;
358 
359  //Read absolute value
360  int value = c - '0';
361  while(u_is_digit(c = u_fgetc(f)))
362  value = value * 10 + (c - '0');
363 
364  //Set sign
365  value *= sign;
366  return value;
367  }

369  void add_transitions(state_ref_array &called_states, ul_state_ref_x_transition_indexes_vector_ref_vector &stv, tag_ref_vector &trv)
370  {
371  typename ul_state_ref_x_transition_indexes_vector_ref_vector::iterator i;
372  typename transition_indexes_vector::iterator j;
373 
374  //For each state
375  for (i = stv.begin(); i != stv.end(); ++i)
376  {
377 #ifdef TRACE
378  std::wcout << "Adding outgoing transitions from state " << static_cast<state_const_ref>(i->q_ref)->wlabel(L'q') << std::endl;
379 #endif
380  //And for each outgoing transition
381  for (j = i->tv_ref->begin(); j != i->tv_ref->end(); ++j)
382  {
383 #ifdef TRACE
384  std::wcout << "-> " << static_cast<state_const_ref>(stv[j->target_index].q_ref)->wlabel(L'q') << " with label " << j->label_index << std::endl;
385 #endif
386  //Add call-transition (negative tag index corresponds to a called state)
387  if (j->label_index < 0)
388  i->q_ref->add_call_transition(called_states[-1 - j->label_index], stv[j->target_index].q_ref);
389  else
390  {
391  //Add epsilon transition (tag input lexical mask is NULL)
392  switch (trv[j->label_index]->input->type)
393  {
394  case CONSUMING:
395  //There is a defined output
396  if (trv[j->label_index]->has_non_default_output)
397  i->q_ref->add_translating_transition(trv[j->label_index]->input->lexmask_ref, trv[j->label_index]->output, stv[j->target_index].q_ref);
398  //Else output is empty
399  else
400  i->q_ref->add_deleting_transition(trv[j->label_index]->input->lexmask_ref, stv[j->target_index].q_ref);
401  break;
402  case EPSILON:
403  if (trv[j->label_index]->has_non_default_output)
404  i->q_ref->add_inserting_transition(trv[j->label_index]->output, stv[j->target_index].q_ref);
405  else
406  i->q_ref->add_epsilon_transition(stv[j->target_index].q_ref);
407  break;
408  case NO_BLANK_EPSILON:
409  if (trv[j->label_index]->has_non_default_output)
410  i->q_ref->add_no_blank_inserting_transition(trv[j->label_index]->output, stv[j->target_index].q_ref);
411  else
412  i->q_ref->add_no_blank_epsilon_transition(stv[j->target_index].q_ref);
413  break;
414  case BLANK_EPSILON:
415  if (trv[j->label_index]->has_non_default_output)
416  i->q_ref->add_blank_inserting_transition(trv[j->label_index]->output, stv[j->target_index].q_ref);
417  else
418  i->q_ref->add_blank_epsilon_transition(stv[j->target_index].q_ref);
419  break;
420  default: fatal_error("Trying to create transition with invalid coarse lexmask type\n");
421  }
422  }
423  }
424  }
