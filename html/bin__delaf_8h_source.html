68  //State class just helps to interpretate a sequence of bytes as a FST state
69  //Byte 0: negation of first bit = final flag; Remaining bits are the most significant of the state transition count
70  //Byte 1: less significant bits of transition count
71  //Byte 2-4: if final, these 3 bytes are the word properties index (begining with de most significant byte); otherwise transitions start here
72  class state
73  {
74  private:
75  const unsigned char byte0, byte1, byte2, byte3, byte4;
76 
77  //Packed states are not built; we only interpretate a byte sequence as a packed state
78  //Thus, call to state constructor is forbidden by declaring it as private
79  state(): byte0(0), byte1(0), byte2(0), byte3(0), byte4(0)
80  {}

103  const_outgoing_transition_iterator outgoing_transition_begin() const
104  {

118  //Returns the outgoing transition at the specified index
119  //Notice that we do not check if the index is out of bounds
120  const outgoing_transition& get_outgoing_transition(std::size_t index) const

123  //Binary searches the outgoing transition corresponding to the specified character
124  //and returns the reference of its target state, or a NULL reference if the transition
125  //does not exist
126  std::size_t get_target_state_offset(const unichar c) const
127  {
128  const outgoing_transition* current_transition_ref;
129  std::size_t middle_transition_index;
130  std::size_t lower_transition_index(0);
131  std::size_t upper_transition_index(get_outgoing_transition_count());
132  unichar current_unichar;
133 
134  //While there are transitions to be searched
135  while (lower_transition_index != upper_transition_index)
136  {
137  //Get current middle transition index
138  middle_transition_index = (lower_transition_index + upper_transition_index) / 2;
139  current_transition_ref = &get_outgoing_transition(middle_transition_index);
140  current_unichar = current_transition_ref->get_label();
141 // std::wcout << "Transition char, lbound, cbound, ubound, offset: " << current_unichar << L" " << lower_transition_index << L" " << middle_transition_index << L" " << upper_transition_index << L" " << ((std::size_t)current_transition_ref) - ((std::size_t)this) << std::endl;
142  //If the current transition has a less unichar, increase the lower transition bound
143  if (current_unichar < c)
144  lower_transition_index = middle_transition_index + 1;
145  //Else if the current transition has a greater unichar, decrease the upper transition bound
146  else if (c < current_unichar)
147  upper_transition_index = middle_transition_index;
148  //Else the matching transition has been found; return the corresponding target state
149  else return current_transition_ref->get_target_state_offset();
150  }
151  //No matching transition found; return NULL state reference
152  return std::numeric_limits<std::size_t>::max();
153  }

156  class const_depth_first_iterator
157  {
158  friend class bin_delaf;
159  public:
160  typedef std::pair<const_outgoing_transition_iterator, const_outgoing_transition_iterator> const_outgoing_transition_iterator_pair;
161  typedef std::deque<const_outgoing_transition_iterator_pair> backtracking_stack;
162  protected:
163  const bin_delaf &dico;
164  backtracking_stack bs;
165 
166  //End iterator constructor
167  const_depth_first_iterator(const bin_delaf &dico_): dico(dico_), bs()
168  {}
169 
170  //Begin iterator constructor
171  const_depth_first_iterator(const bin_delaf &dico_, const state &initial_state): dico(dico_), bs()
172  {
173  if (initial_state.get_outgoing_transition_count())
174  bs.push_back(const_outgoing_transition_iterator_pair(initial_state.outgoing_transition_begin(), initial_state.outgoing_transition_end()));
175  }
176 
177  public:
178  //This is the end iterator once the backtracking stack is empty
179  bool is_end()
180  { return bs.empty(); }
181 
182  //Increment iterator
183  const_depth_first_iterator& operator++()
184  {
185  if (is_end())
186  return *this;
187  //Get next inner state (the state pointed by the transition at the top of the backtracking stack)
188  const state& in_state = dico.offset_to_state(bs.back().first->get_target_state_offset());
189  //If inner state has transitions, push its first transition and we are done
190  if (in_state.get_outgoing_transition_count())
191  {
192 // std::wcout << "Goin" << std::endl;
193  bs.push_back(const_outgoing_transition_iterator_pair(in_state.outgoing_transition_begin(), in_state.outgoing_transition_end()));
194  return *this;
195  }
196  //Else we have to recursively pop transition iterators up to the first one that is not an end iterator, and increment it
197  //or if we run out of transition iterators, return the end iterator
198  do
199  {
200  //Increment top transition iterator
201  ++bs.back().first;
202 // std::wcout << "Inctop" << std::endl;
203  //If it is not an end iterator, we are done
204  if (bs.back().first != bs.back().second)
205  return *this;
206  //Otherwise pop it and go for the next one in the top of the backtracking stack,
207  //or if there are no more return the end iterator
208  bs.pop_back();
209 // std::wcout << "Poptop" << std::endl;
210  } while (!is_end());
211  return *this;
212  }
213 
214  bool operator==(const const_depth_first_iterator &cdfi) const
215  { return bs == cdfi.bs; }
216 
217  bool operator!=(const const_depth_first_iterator &cdfi) const
218  { return !(*this == cdfi); }

335  //Returns the offset of the last state reached by consuming the specified word;
336  //If there is no such state (the word is unknown), the maximum offset is returned
337  template<typename UCharIterator>
338  std::size_t get_state_offset(UCharIterator begin, UCharIterator end) const
339  {
340  std::size_t current_state_offset(get_initial_state_offset());
341  //For each char of the word to recognize
342  for (; begin != end; ++begin)
343  {
344  //Consume the next char and retrieve the next state offset
345  current_state_offset = offset_to_state(current_state_offset).get_target_state_offset(*begin);
346  //If there is no next state, the word is unknown: return maximum state offset
347  if (current_state_offset == std::numeric_limits<std::size_t>::max())
348  break;
349  }
350  //All word chars have been consumed, return current state offset
351  //Notice that if the last consumed character did not lead to any state,
352  //we are returning here the maximum state offset
353  return current_state_offset;
354  }
