129  bool operator< (const execution_state& x2) const
130  {
131  if (q < x2.q)
132  return true;
133  if (x2.q < q)
134  return false;
135  //We don't care here about lexicographical comparison, we simply compare the object addresses in order to optimize comparison
136  if (state_const_ref_stack_less()(pi, x2.pi))
137  return true;

143  std::ptrdiff_t compare_3w(const execution_state& x2) const
144  {
145  //We don't care here about lexicographical comparison, we simply compare the object addresses in order to optimize comparison
146  std::ptrdiff_t result(reinterpret_cast<std::ptrdiff_t>(q) - reinterpret_cast<std::ptrdiff_t>(x2.q));

148  return result;

260  //Increment parsing chart with the epsilon-closure of the last active set of execution states
261  template<typename ExtraInsertOp>
262  void eclosure(SourceRef in, bool next_token_isnt_white_separated, ExtraInsertOp op)
263  {
264  //While there are execution states pending to be processed
265  while (!e.empty())
266  {
267  //Pop next pending active execution state (q, b, lambda, q_h, j)
268  const execution_state &x_s(*e.front());
269  e.pop();
270 
271  //Process epsilon and inserting transitions
272  process_epsilon_transitions(x_s, x_s.q->outgoing_epsilon_transitions.begin(), x_s.q->outgoing_epsilon_transitions.end(), op);
273  process_inserting_transitions(x_s, x_s.q->outgoing_inserting_transitions.begin(), x_s.q->outgoing_inserting_transitions.end(), in, op);
274 
275  //Process no-blank epsilon and inserting transitions
276  //if there are no whites between the current (or input begin) and the next token (or input end)
277  if (next_token_isnt_white_separated)
278  {
279 #ifdef TRACE
280  process_epsilon_transitions(x_s, x_s.q->outgoing_no_blank_epsilon_transitions.begin(), x_s.q->outgoing_no_blank_epsilon_transitions.end(), op, L"no-blank ");
281  process_inserting_transitions(x_s, x_s.q->outgoing_no_blank_inserting_transitions.begin(), x_s.q->outgoing_no_blank_inserting_transitions.end(), in, op, L"no-blank ");
282 #else
283  process_epsilon_transitions(x_s, x_s.q->outgoing_no_blank_epsilon_transitions.begin(), x_s.q->outgoing_no_blank_epsilon_transitions.end(), op);
284  process_inserting_transitions(x_s, x_s.q->outgoing_no_blank_inserting_transitions.begin(), x_s.q->outgoing_no_blank_inserting_transitions.end(), in, op);
285 #endif
286  }
287 
288  //Process blank epsilon and inserting transitions otherwise
289  else
290  {
291 #ifdef TRACE
292  process_epsilon_transitions(x_s, x_s.q->outgoing_blank_epsilon_transitions.begin(), x_s.q->outgoing_blank_epsilon_transitions.end(), op, L"blank ");
293  process_inserting_transitions(x_s, x_s.q->outgoing_blank_inserting_transitions.begin(), x_s.q->outgoing_blank_inserting_transitions.end(), in, op, L"blank ");
294 #else
295  process_epsilon_transitions(x_s, x_s.q->outgoing_blank_epsilon_transitions.begin(), x_s.q->outgoing_blank_epsilon_transitions.end(), op);
296  process_inserting_transitions(x_s, x_s.q->outgoing_blank_inserting_transitions.begin(), x_s.q->outgoing_blank_inserting_transitions.end(), in, op);
297 #endif
298  }
299 
300  //Process call-transitions; Earley's predictor
301  for (outgoing_call_transition_set_iterator call_transition_begin(x_s.q->outgoing_call_transitions.begin()); call_transition_begin != x_s.q->outgoing_call_transitions.end(); ++call_transition_begin)
302  {
303  state_const_ref target(static_cast<state_const_ref>(call_transition_begin->target));
304  std::pair<execution_state_set_iterator, bool> result(v.insert(execution_state(static_cast<state_const_ref>(call_transition_begin->called), x_s.b, x_s.pi + target)));
305  if (result.second)
306  {
307 #ifdef TRACE
308  result.first->serialize(std::wcout) << L" push" << std::endl;
309 #endif
310  e.push(&(*result.first));
311  op(*result.first);
312  }
313  }
314 
315  //Process pop-transition; Earley's completer
316  if (x_s.q->is_final() && !x_s.pi.empty())
317  {
318  std::pair<execution_state_set_iterator, bool> result(v.insert(execution_state(static_cast<state_const_ref>(x_s.pi.back()), x_s.b, x_s.pi.prefix())));
319  if (result.second)
320  {
321 #ifdef TRACE
322  result.first->serialize(std::wcout) << L" pop" << std::endl;
323 #endif
324  e.push(&(*result.first));
325  op(*result.first);
326  }
327  }
328  }
329  }

332  //Earley scanner: adds a new item to the parsing chart and initializes it with the attainable set of execution states from the former last chart item set of execution states through the consumption of an input symbol sigma
333  template<typename ExtraInsertOp>
334  execution_state_set translate_symbol(SourceRef in_i, ExtraInsertOp op)
335  {
336  execution_state_set w;
337 
338  //For each (q, b, lambda, q_h, i) in v
339  for (execution_state_set_const_iterator x_s_it(v.begin()); x_s_it != v.end(); ++x_s_it)
340  {
341  //For each (q', lambda) = delta(q, (sigma, epsilon))
342  //Process deleting transitions
343  for (outgoing_deleting_transition_set_const_iterator dt_it(x_s_it->q->outgoing_deleting_transitions.begin()); dt_it != x_s_it->q->outgoing_deleting_transitions.end(); ++dt_it)
344  {
345  if (input_match(*in_i, dt_it->input))
346  {
347  //w = w union (q', b, lambda, q_h, i)
348  std::pair<execution_state_set_iterator, bool> result(w.insert(execution_state(static_cast<state_const_ref>(dt_it->target), x_s_it->b, x_s_it->pi)));
349  if (result.second)
350  {
351 #ifdef TRACE
352  result.first->serialize(std::wcout) << L" (";
353  input_serializer()(std::wcout, *in_i) << L" : <E>)" << std::endl;
354 #endif
355  e.push(&(*result.first));
356  op(*result.first);
357  }
358  }
359  }
360  //For each (q', lambda) = delta(q, (sigma, g))
361  //Process translating transitions
362  for (outgoing_translating_transition_set_const_iterator tt_it(x_s_it->q->outgoing_translating_transitions.begin()); tt_it != x_s_it->q->outgoing_translating_transitions.end(); ++tt_it)
363  {
364  if (input_match(*in_i, tt_it->input))
365  {
366  //w = w union (q', bg, lambda, q_h, i)
367  blackboard b_t(x_s_it->b);
368  std::pair<execution_state_set_iterator, bool> result(w.insert(execution_state(static_cast<state_const_ref>(tt_it->target), gamma(b_t, in_i, tt_it->output), x_s_it->pi)));
369  if (result.second)
370  {
371 #ifdef TRACE
372  result.first->serialize(std::wcout) << L" (";
373  input_serializer()(std::wcout, *in_i) << L" : ";
374  tag_output_serializer()(std::wcout, tt_it->output) << L')' << std::endl;
375 #endif
376  e.push(&(*result.first));
377  op(*result.first);
378  }
379  }
380  }
381  }
382  return w;
383  }

401  //Compute the r-translations of the input range [input_being, input_end) and add them to the set t
402  blackboard_set& operator()(const machine& grammar, SourceRef input_begin, SourceRef input_end, bool hasnt_white_at_begin, bool hasnt_white_at_end, blackboard_set &t, const blackboard &empty_blackboard = blackboard())
403  {
404 #ifdef TRACE
405  std::wcout << L"Begin parsing\n";
406 #endif
407  //Initialize pool of state sequences
408  state_const_ref_stack_pool scrsp;
409 
410  //Initialize sets of execution states V and V^prime
411  v.clear();
412 
413  no_op_on_execution_state no_op;
414  acceptor_execution_state_inserter ins_op(t);
415 
416  //Build V_0 as the closure of Q X {b_emptyset} X {lambda};
417  //If empty input, build initial and final V[0]
418  if (input_begin == input_end)
419  {
420  build_initial_ses(grammar.initial_state(), scrsp.empty(), hasnt_white_at_begin, ins_op, empty_blackboard);
421  //First token is white separated if there are trailing whites at the beginning
422  eclosure(input_begin, hasnt_white_at_begin, ins_op);
423  }
424  //Else build initial V[0] and the remaining V[i]
425  else
426  {
427  build_initial_ses(grammar.initial_state(), scrsp.empty(), hasnt_white_at_begin, no_op, empty_blackboard);
428  //First token is white separated if there are trailing whites at the beginning
429  eclosure(input_begin, hasnt_white_at_begin, no_op);
430 #ifdef TRACE
431  std::size_t idx(0);
432 #endif
433  SourceRef input_begin_plus_1(input_begin);
434  --input_end;
435  //While the last set of execution states is not empty and there are input symbols left, compute the next set of execution states
436  while (!v.empty() && input_begin != input_end)
437  {
438 #ifdef TRACE
439  std::wcout << L"----- V[" << static_cast<int>(idx) + 1 << L"] -----" << std::endl;
440 #endif
441  ++input_begin_plus_1;
442  execution_state_set w(translate_symbol(input_begin, no_op));
443 #ifdef TRACE
444  ++idx;
445 #endif
446  v.swap(w);
447  eclosure(input_begin_plus_1, (*input_begin)->end == (*input_begin_plus_1)->begin, no_op);
448  ++input_begin;
449  }
450  ++input_end;
451 
452  //Compute last set of execution states
453  if (!v.empty())
454  {
455 #ifdef TRACE
456  std::wcout << L"Last----- V[" << static_cast<int>(idx) + 1 << L"] -----" << std::endl;
457 #endif
458  ++input_begin_plus_1;
459  execution_state_set w(translate_symbol(input_begin, ins_op));
460 #ifdef TRACE
461  ++idx;
462 #endif
463  v.swap(w);
464  //Next token is the input end... it is white separated if there are trailing whites at the end
465  eclosure(input_begin_plus_1, hasnt_white_at_end, ins_op);
466  ++input_begin;
467  }
468  }
469 #ifdef TRACE
470  std::wcout << L"End parsing\n";
471 #endif
472  return t;
