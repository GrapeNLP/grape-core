62  n_term read_n_term_in_grammar(FILE *grammar_file, n_term_trie_string_ref label_ref)
63  {
64  unichar c(u_fgetc(grammar_file));
65  while (!feof(grammar_file) && c != '\n' && c != '^' && c != '_')
66  {
67  label_ref = &(*label_ref + c);
68  c = u_fgetc(grammar_file);
69  }
70  if (feof(grammar_file))
71  fatal_error("Unexpected EOF while reading non-terminal\n");
72  if (c == '\n')
73  fatal_error("Unexpected EOL while reading non-terminal\n");
74  bool is_pos = (c == '_');
75  if (!is_pos)
76  {
77  //Skip 'g'
78  c = u_fgetc(grammar_file);
79  if (c != 'g')
80  fatal_error("Expected 'g' but got '%c' while reading non-terminal\n", c);
81  //Skip '_'
82  c = u_fgetc(grammar_file);
83  if (c != '_')
84  fatal_error("Expected '_' but got '%c' while reading non-terminal\n", c);
85  }
86  idx_type idx;
87  u_fread_unsigned_int(grammar_file, idx);
88  return n_term(label_ref, idx, is_pos);
89  }

102  void read_grammar_rule(FILE *grammar_file, algebra &grammar, unichar &c)
103  {
104  n_term head(read_n_term_in_grammar(grammar_file, grammar.n_term_dico, c));
105  //Skip '-'
106  u_fgetc(grammar_file);
107  //Skip '>'
108  u_fgetc(grammar_file);
109  //Skip ' '
110  u_fgetc(grammar_file);
111  n_term body_l(read_n_term_in_grammar(grammar_file, grammar.n_term_dico));
112  c = u_fgetc(grammar_file);
113  if (u_is_digit(c))
114  {
115  probability p;
116  //Also skips EOL
117  c = u_fread_float(grammar_file, p, c);
118  unary_rule un(head, body_l, p);
119  if (head != body_l)
120  {
121  if (!grammar.add_unary_rule(head, body_l, p))
122  {
123  std::wcerr << "Repeated unary rule ";
124  un.serialize(std::wcerr) << std::endl;
125  exit(1);
126  }
127  }
128  else
129  {
130  std::wcerr << "Warning: unitary rule \"";
131  un.serialize(std::wcerr) << "\" omitted\n";
132  }
133  return;
134  }
135  n_term body_r(read_n_term_in_grammar(grammar_file, grammar.n_term_dico, c));
136  probability p;
137  //Also skips EOL
138  c = u_fgetc(grammar_file);
139  c = u_fread_float(grammar_file, p, c);
140  binary_rule bn(head, body_l, body_r, p);
141  if (!grammar.add_binary_rule(head, body_l, body_r, p))
142  {
143  std::wcerr << "Repeated binary rule ";
144  bn.serialize(std::wcerr) << std::endl;
145  exit(1);
146  }
147  }

184  void read_tokens(FILE *lexicon_file, lexicon_trie_string_ref epsilon, std::list<lexicon_trie_string_ref> &tokens)
185  {
186  unichar c(u_fgetc(lexicon_file));
187  lexicon_trie_string_ref s_ref;
188  while (!feof(lexicon_file) && c != ' ' && c != '\n')
189  {
190  if (u_is_letter(c))
191  {
192  s_ref = epsilon;
193  do s_ref = &(s_ref->concat(c));
194  while (!feof(lexicon_file) && u_is_letter(c = u_fgetc(lexicon_file)));
195  s_ref->data.is_token = true;
196  tokens.push_back(s_ref);
197  }
198  else
199  {
200  s_ref = &(epsilon->concat(c));
201  s_ref->data.is_token = true;
202  tokens.push_back(s_ref);
203  c = u_fgetc(lexicon_file);
204  }
205  }
206  if (feof(lexicon_file))
207  fatal_error("Unexpected EOF while reading term\n");
208  if (c == '\n')
209  fatal_error("Unexpected EOL while reading term\n");
210  }

212  void read_lexicon_probabilities(FILE *lexicon_file, array<Probability> &pp, unichar &c)
213  {
214  //Skip '['
215  c = u_fgetc(lexicon_file);
216  if (c != '[')
217  fatal_error("Expected '[' but got %c\n", c);
218  c = u_fgetc(lexicon_file);
219  if (!u_is_float_first(c))
220  fatal_error("Expected first character of of first lexical probability but got '%c'\n", c);
221 
222  typename array<Probability>::iterator it(pp.begin());
223  c = u_fread_float(lexicon_file, *it, c);
224  for (++it; it != pp.end(); ++it)
225  {
226  if (c != ',')
227  fatal_error("Expected ',' after lexical probability but got '%c'\n", c);
228  c = u_fgetc(lexicon_file);
229  if (c != ' ')
230  fatal_error("Expected space after comma after lexical probability but got '%c'\n", c);
231  c = u_fgetc(lexicon_file);
232  if (!u_is_float_first(c))
233  fatal_error("Expected first digit of lexical probability but got '%c'\n", c);
234  c = u_fread_float(lexicon_file, *it, c);
235  if (feof(lexicon_file))
236  fatal_error("Unexpected EOF while reading lexical probabilities\n");
237  }
238  if (c != ']')
239  fatal_error("Expected ']' closing the list of lexical probabilities but got %c\n", c);
240  }

242  void read_lexical_rule(FILE *lexicon_file, algebra &grammar, lexicon_trie_string_ref unk_prefix_ref, unichar &c)
243  {
244  n_term_trie_string_ref head_ref(read_n_term_in_lexicon(lexicon_file, grammar.n_term_dico, c));
245  std::list<lexicon_trie_string_ref> token_refs;
246  read_tokens(lexicon_file, &grammar.lexicon_dico.epsilon(), token_refs);
247  array<Probability> pp(head_ref->data.count);
248  read_lexicon_probabilities(lexicon_file, pp, c);
249  //Ignore unknown word rules
250  if (token_refs.front() != unk_prefix_ref)
251  {
252  std::pair<lexical_rule_set_iterator, bool> result(grammar.add_lexical_rule(head_ref, token_refs, pp));
253  if (!result.second)
254  {
255  result.first->serialize(std::wcerr << "Repeated lexicon rule: ");
256  exit(1);
257  }
258  }

262  {
263  unichar c(u_fgetc(lexicon_file));
264  lexicon_trie_string_ref unk_prefix(&(grammar.lexicon_dico.epsilon().concat('U').concat('N').concat('K')));
