58  u_array lexmask_code(uls.size());

60  //Word mask
61  if (u_is_letter(lexmask_code[1]))
62  {
63  if (lexmask_code[0] == case_sensitive_char)
64  {
65 #ifdef TRACE
66  std::wcout << L"Creating case sensitive mask\n";
67 #endif
68  return new lexmask_and_coarse_type<InputIterator>(new case_sensitive_word_mask<InputIterator>(lexmask_code.begin() + 1, lexmask_code.size() - 1), CONSUMING);

71  std::wcout << L"Creating case insensitive mask\n";
72 #endif

75 
76  //Meta mask
77  if (lexmask_code[1] == meta_left_delimiter_char && lexmask_code[0] == case_insensitive_char && lexmask_code.size() > 2 && lexmask_code[2] != in_out_separator_char)
78  {
79  if (lexmask_code.back() != meta_right_delimiter_char)
80  fatal_error("Unterminated meta mask\n");
81  u_array::const_iterator meta_code_begin = lexmask_code.begin() + 2;
82  u_array::const_iterator meta_code_end = lexmask_code.end() - 1;
83 
84  u_trie_string::ref meta_code(the_meta_code_dico.get(meta_code_begin, meta_code_end));
85  //Epsilon
86  if (meta_code == the_meta_code_dico.epsilon_meta_code)
87  {
88 #ifdef TRACE
89  std::wcout << L"Creating epsilon meta mask\n";
90 #endif
91  return new lexmask_and_coarse_type<InputIterator>(NULL, EPSILON);
92  }
93  //Token meta mask
94  if (meta_code == the_meta_code_dico.token_meta_code)
95  {
96 #ifdef TRACE
97  std::wcout << L"Creating token meta mask\n";
98 #endif
99  return new lexmask_and_coarse_type<InputIterator>(new token_meta_mask<InputIterator>(), CONSUMING);
100  }
101  //Word meta mask
102  if (meta_code == the_meta_code_dico.word_meta_code)
103  {
104 #ifdef TRACE
105  std::wcout << L"Creating word meta mask\n";
106 #endif

108  }

116  }
117  //Upper word meta mask
118  if (meta_code == the_meta_code_dico.upper_word_meta_code)
119  {
120 #ifdef TRACE
121  std::wcout << L"Creating upper word meta mask\n";
122 #endif
123  return new lexmask_and_coarse_type<InputIterator>(new upper_word_meta_mask<InputIterator>(), CONSUMING);
124  }
125  //Not an upper word meta mask

164  }

216 #endif
