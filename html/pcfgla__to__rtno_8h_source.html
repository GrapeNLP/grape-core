128  binary_rule_set_iterator br_first(br_current);

135  state_ref called_state_ref;
136  u_trie_string_ref out_close_tag_string_ref(&out_tag_trie.epsilon());

143  out_close_tag_string_ref = &(out_close_tag_string_ref->concat(')'));

272  void add_path_for_tokens(machine &rtno_grammar, state_ref start, const std::list<lexicon_trie_string_ref> &tokens, state_ref end) const
273  {
274  typename std::list<lexicon_trie_string_ref>::const_iterator it(tokens.begin());
275  typename std::list<lexicon_trie_string_ref>::const_iterator last_it(tokens.end());
276  --last_it;
277  state_ref target;
278  while (it != last_it)
279  {
280  target = rtno_grammar.create_state(false);
281  start->add_deleting_transition((*it)->data.lmask_cref, target);
282  start = rtno_grammar.create_state(false);
283  target->add_no_blank_epsilon_transition(start);
284  ++it;
285  }
286  start->add_deleting_transition((*it)->data.lmask_cref, end);
287  }

289  void add_lexical_rule_for_pos(const algebra& pcfgla_grammar, machine &rtno_grammar, u_trie &out_tag_trie, lexical_rule_set_iterator &lr_current) const
290  {
291  IdxType count(lr_current->head->data.count);
292  lexical_rule_set_iterator lr_first(lr_current);
293  u_trie_string_ref out_open_tag_label_ref(get_out_open_tag_label_ref_for_pos(lr_first->head, out_tag_trie));
294  u_trie_string_ref out_close_tag_string_ref(&(out_tag_trie.epsilon().concat(')')));
295  tag_output out_open_tag(out_open_tag_label_ref, static_cast<Probability>(1.0));
296  array<state_ref> before_word_state_refs(count);
297  array<state_ref> acceptor_state_refs(count);
298  IdxType idx(0);
299  for (; idx < count; ++idx)
300  {
301  before_word_state_refs[idx] = rtno_grammar.create_state(false);
302  acceptor_state_refs[idx] = rtno_grammar.create_state(true);
303  get_subinitial_state_ref_for_pos(lr_first->head, idx)->add_inserting_transition(out_open_tag, before_word_state_refs[idx]);
304  }
305  state_ref before_probability_state_ref;
306  do
307  {
308  for (idx = 0; idx < count; ++idx)
309  {
310  before_probability_state_ref = rtno_grammar.create_state(false);
311  add_path_for_tokens(rtno_grammar, before_word_state_refs[idx], lr_current->tokens, before_probability_state_ref);
312  before_probability_state_ref->add_inserting_transition(tag_output(out_close_tag_string_ref, lr_current->pp[idx]), acceptor_state_refs[idx]);
313  }
314  ++lr_current;
315  } while (lr_current != pcfgla_grammar.lexical_rules.end() && lr_first->head == lr_current->head);
316  }

318  void create_subinitial_states(const algebra &pcfgla_grammar, machine &rtno_grammar) const
319  {
320  rtno_grammar.create_state(false); //global initial state
321  n_term_trie_string_ref axiom_label_ref(pcfgla_grammar.get_axiom_label_ref());
322  IdxType idx;
323  for (n_term_trie_iterator it(pcfgla_grammar.n_term_dico.begin()); it != pcfgla_grammar.n_term_dico.end(); ++it)
324  {
325  if (it->data.count)
326  {
327  it->data.subinitial_state_refs = new state_ref[it->data.count];
328  idx = 0;
329  do
330  {
331  if (&(*it) == axiom_label_ref)
332  it->data.subinitial_state_refs[idx] = rtno_grammar.initial_state();
333  else it->data.subinitial_state_refs[idx] = rtno_grammar.create_state(false);
334  ++idx;
335  } while (idx < it->data.count);
336  }
337  }

340  void create_lexical_masks(const algebra &pcfgla_grammar) const
341  {
342  for (lexicon_trie_iterator it(pcfgla_grammar.lexicon_dico.begin()); it != pcfgla_grammar.lexicon_dico.end(); ++it)
343  {
344  if (it->data.is_token)
345  {
346  if (u_is_letter(it->back()))
347  it->data.lmask_cref = new case_sensitive_word_mask<InputIterator>(it->begin(), it->end(), it->size());
348  else it->data.lmask_cref = new symbol_mask<InputIterator>(it->back());
349  }

353  machine& operator() (const algebra &pcfgla_grammar, machine &rtno_grammar, u_trie &out_tag_trie, u_dico_word_meta_mask_factory<InputIterator> &dico_lexmask_factory) const
354  {
355  create_subinitial_states(pcfgla_grammar, rtno_grammar);
356  create_lexical_masks(pcfgla_grammar);
357  binary_rule_set_iterator br_it(pcfgla_grammar.binary_rules.begin());
358  binary_rule_set_iterator br_it_end(pcfgla_grammar.binary_rules.end());
359  unary_rule_set_iterator ur_it(pcfgla_grammar.unary_rules.begin());
360  unary_rule_set_iterator ur_it_end(pcfgla_grammar.unary_rules.end());
361 
362  while (br_it != br_it_end || ur_it != ur_it_end)
363  {
364  //Add binary rules for head not in unary rules
365  if (ur_it == ur_it_end || (br_it != br_it_end && br_it->head < ur_it->head))
366  add_binary_rules_for_head(pcfgla_grammar, rtno_grammar, out_tag_trie, br_it);
367  //Add unary rules with head not in binary rules
368  else if (br_it == br_it_end || (ur_it != ur_it_end && ur_it->head < br_it->head))
369  add_unary_rules_for_head(pcfgla_grammar, rtno_grammar, out_tag_trie, ur_it, ur_it == pcfgla_grammar.axiom_unary_rules_first);
370  //Add binary & unary rules with the same head
371  else add_binary_and_unary_rules_for_head(pcfgla_grammar, rtno_grammar, out_tag_trie, br_it, ur_it);
372  }
373 
374  lexical_rule_set_iterator lr_it(pcfgla_grammar.lexical_rules.begin());
375  lexical_rule_set_iterator lr_it_end(pcfgla_grammar.lexical_rules.end());
376  while (lr_it != lr_it_end)
377  add_lexical_rule_for_pos(pcfgla_grammar, rtno_grammar, out_tag_trie, lr_it);
378  return rtno_grammar;
379  }
