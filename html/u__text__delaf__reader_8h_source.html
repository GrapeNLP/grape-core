84  template<typename CaseNormalizer>
85  bool u_read_entry(FILE *f, std::size_t entry_count, u_text_delaf<CaseNormalizer> &d)
86  {
87  //Load mandatory lemma
88  //A whole DELAF entry should be present, starting with the lemma, or nothing
89  //if we are already at the end of the dictionary
90  typename u_text_delaf<CaseNormalizer>::u_lemma_string::ref lemma(&d.lemma_to_entry_index.epsilon());
91  unichar next_char = u_read_lemma<CaseNormalizer>(f, lemma);
92  //Skip comment lines and read the following DELAF entry, or return if end of dico reached
93  while (lemma == &d.lemma_to_entry_index.epsilon() && next_char == comment_separator)
94  {
95  //Skip comment line
96  u_fskip_line(f);
97  //End of dico reached
98  if (feof(f))
99  return false;
100  //Read next dico comment line or entry
101  next_char = u_read_lemma<CaseNormalizer>(f, lemma);
102  }
103  //If the lemma is empty but this is not a comment line
104  if (lemma == &d.lemma_to_entry_index.epsilon())
105  {
106  //We are at the end of the dico
107  if (feof(f))
108  return false;
109  //Or we are at the end of line just before the end of the dictionary
110  while (u_is_white(next_char))
111  next_char = u_fgetc(f);
112  //Otherwise the DELAF entry is incomplete
113  if (!feof(f))
114  fatal_error("DELAF entry %d truncated after lemma\n", static_cast<int>(entry_count));
115  return false;
116  }
117 
118  //Load mandatory canonical form
119  //The canonical form is mandatory but may be empty;
120  //In that case, the lemma is also the canonical form
121  typename u_text_delaf<CaseNormalizer>::u_lemma_string::ref canonical_form(&d.lemma_to_entry_index.epsilon());
122  next_char = u_read_canonical_form<CaseNormalizer>(f, canonical_form);
123  //After the canonical form there must be the separator between it and the POS;
124  //otherwise the DELAF entry is incomplete
125  if (feof(f) || next_char != canonical_form_separator)
126  fatal_error("DELAF entry %d truncated after canonical form\n", static_cast<int>(entry_count));
127  //If the canonical form is empty, the lemma is also the canonical form
128  if (canonical_form == &d.lemma_to_entry_index.epsilon())
129  canonical_form = lemma;
130 
131  //Load mandatory part-of-speech, with is processed as a semantic trait
132  ust_string::ref pos(&d.ppooss.epsilon());
133  next_char = u_read_semantic_trait(f, pos);
134  //After the POS we may find an EOL, an EOF, a semantic trait separator or a flexional trait separator
135  //The semantic traits, flexional traits and clitic pronouns are optional
136  //Anyway, the POS should not be empty
137  if (pos == &d.ppooss.epsilon())
138  fatal_error("DELAF entry %d having an empty POS\n", static_cast<int>(entry_count));
139 
140  typename u_text_delaf<CaseNormalizer>::u_entry *entry_ref(new typename u_text_delaf<CaseNormalizer>::u_entry(*lemma, *canonical_form, *pos));
141 
142  //Load optional semantic traits
143  //After the POS we may find a sequence of zero, one or more semantic traits
144  if (next_char == semantic_traits_separator)
145  next_char = u_read_semantic_traits<CaseNormalizer>(f, entry_count, d, entry_ref->semantic_traits);
146 
147  //Load optional flexional traits
148  //After the sequence of zero, one or more semantic traits, we may find a sequence of zero, one or more flexional traits
149  if (next_char == flexional_traits_separator)
150  {
151  next_char = u_read_possible_flexional_traits<CaseNormalizer>(f, entry_count, d, entry_ref->possible_flexional_traits);
152  //Load clitic pronouns
153  //If there are flexional traits specified, we may found zero, one or more clitic pronouns afterwards
154  if (next_char == clitic_pronouns_separator)
155  next_char = u_read_clitic_pronouns<CaseNormalizer>(f, entry_count, d, entry_ref->clitic_pronouns);
156  }
157 
158  if (next_char == comment_separator)
159  u_fskip_line(f);
160 
161  //If the lemma string did not have already any DELAF entry, initialize its entry list
162  if (!lemma->data)
163  lemma->data = new typename u_text_delaf<CaseNormalizer>::u_entry_list;
164  //Otherwise, check for duplicated entries and fusion the entry to a duplicate
165  else
166  {
167  typename u_text_delaf<CaseNormalizer>::u_entry_list::iterator i(lemma->data->begin());
168  for (; i != lemma->data->end(); ++i)
169  {
170  if ((*i)->canonical_form == entry_ref->canonical_form && (*i)->pos == entry_ref->pos && (*i)->semantic_traits == entry_ref->semantic_traits)
171  {
172  entry_ref->possible_flexional_traits.transfer((*i)->possible_flexional_traits);
173  return true;
174  }
175  }
176  }
177  //If no duplicates found or this is the first entry for the current lemma, just add it
178  lemma->data->push_back(entry_ref);
179  return true;
180  }

183  unichar u_read_semantic_traits(FILE *f, std::size_t entry_count, u_text_delaf<CaseNormalizer> &dico, typename u_text_delaf<CaseNormalizer>::ust_set &semantic_traits)
184  {
185  ust_string::ref semantic_trait;
186  unichar next_char;
187  do
188  {
189  semantic_trait = &dico.semantic_traits.epsilon();
190  next_char = u_read_semantic_trait(f, semantic_trait);
191  if (semantic_trait == &dico.semantic_traits.epsilon())
192  fatal_error("DELAF entry %d having an empty semantic trait\n", static_cast<int>(entry_count));
193  semantic_traits.insert(semantic_trait);
194  } while (!feof(f) && next_char == semantic_traits_separator);
