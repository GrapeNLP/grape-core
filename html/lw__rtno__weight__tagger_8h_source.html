56  machine& operator() (machine &m)
57  {
58 #ifdef TRACE
59  std::wcout << L"Setting ini state outgoing transition weights" << std::endl;
60 #endif
61  lexmask_default_weight<InputIterator, weight> ldw;
62  state_ref_list_iterator s_ref_it;
63  // typename outgoing_epsilon_transition_set::iterator outgoing_epsilon_transition_iterator;
64  typename outgoing_deleting_transition_set::iterator outgoing_deleting_transition_iterator;
65  // typename outgoing_inserting_transition_set::iterator outgoing_inserting_transition_iterator;
66  typename outgoing_translating_transition_set::iterator outgoing_translating_transition_iterator;
67 
68  //For each state
69  for (s_ref_it = m.states.begin(); s_ref_it != m.states.end(); ++s_ref_it)
70  {
71  //Create a new set of translating transitions
72  outgoing_translating_transition_set outgoing_translating_transitions;
73  //And swap the original translating transition set with the pivot
74  (*s_ref_it)->outgoing_translating_transitions.swap(outgoing_translating_transitions);
75  //Fill the new translating transition set with the former translating transitions but
76  //reseting their weights depending on the input lexmask specificity
77  for (outgoing_translating_transition_iterator = outgoing_translating_transitions.begin(); outgoing_translating_transition_iterator != outgoing_translating_transitions.end(); ++outgoing_translating_transition_iterator)
78  {
79  weighted_tag_output to(outgoing_translating_transition_iterator->output);
80  to.w = ldw(*outgoing_translating_transition_iterator->input);
81  (*s_ref_it)->add_translating_transition(outgoing_translating_transition_iterator->input, to, outgoing_translating_transition_iterator->target);
82  }
83 
84  //Transform deleting transitions into translating transitions (generating a weight)
85  //First copy each consuming transition to the set of translating transitions by adding the default
86  //weight as output, then clear the consuming transition set
87  for (outgoing_deleting_transition_iterator = (*s_ref_it)->outgoing_deleting_transitions.begin(); outgoing_deleting_transition_iterator != (*s_ref_it)->outgoing_deleting_transitions.end(); ++outgoing_deleting_transition_iterator)
88  {
89  weighted_tag_output to(ldw(*outgoing_deleting_transition_iterator->input));
90  (*s_ref_it)->add_translating_transition(outgoing_deleting_transition_iterator->input, to, outgoing_deleting_transition_iterator->target);
91  }
92  (*s_ref_it)->outgoing_deleting_transitions.clear();
93  }
94  return m;
95  }
