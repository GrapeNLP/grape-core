348  template<typename ExtraInsertOp>
349  void eclosure(std::size_t idx, SourceRef in, bool next_token_isnt_white_separated, ExtraInsertOp op)
350  {
351  chart_item &v = the_chart[idx];
352  deletable_call_set t;
353 
354  //While there are execution states pending to be processed
355  while (!e.empty())
356  {
357  //Pop next pending active execution state (q, b, lambda, q_h, j)
358  const active_execution_state &x_s(*e.front());
359  e.pop();
360 
361  //Process epsilon and inserting transitions
362  process_epsilon_transitions(x_s, x_s.q->outgoing_epsilon_transitions.begin(), x_s.q->outgoing_epsilon_transitions.end(), v, op);
363  process_inserting_transitions(x_s, x_s.q->outgoing_inserting_transitions.begin(), x_s.q->outgoing_inserting_transitions.end(), v, in, op);
364 
365  //Process no-blank epsilon and inserting transitions
366  //if there are no whites between the current (or input begin) and the next token (or input end)
367  if (next_token_isnt_white_separated)
368  {
369 #ifdef TRACE
370  process_epsilon_transitions(x_s, x_s.q->outgoing_no_blank_epsilon_transitions.begin(), x_s.q->outgoing_no_blank_epsilon_transitions.end(), v, op, L"no-blank ");
371  process_inserting_transitions(x_s, x_s.q->outgoing_no_blank_inserting_transitions.begin(), x_s.q->outgoing_no_blank_inserting_transitions.end(), v, in, op, L"no-blank ");
372 #else
373  process_epsilon_transitions(x_s, x_s.q->outgoing_no_blank_epsilon_transitions.begin(), x_s.q->outgoing_no_blank_epsilon_transitions.end(), v, op);
374  process_inserting_transitions(x_s, x_s.q->outgoing_no_blank_inserting_transitions.begin(), x_s.q->outgoing_no_blank_inserting_transitions.end(), v, in, op);
375 #endif
376  }
377 
378  //Process blank epsilon and inserting transitions otherwise
379  else
380  {
381 #ifdef TRACE
382  process_epsilon_transitions(x_s, x_s.q->outgoing_blank_epsilon_transitions.begin(), x_s.q->outgoing_blank_epsilon_transitions.end(), v, op, L"blank ");
383  process_inserting_transitions(x_s, x_s.q->outgoing_blank_inserting_transitions.begin(), x_s.q->outgoing_blank_inserting_transitions.end(), v, in, op, L"blank ");
384 #else

386  process_inserting_transitions(x_s, x_s.q->outgoing_blank_inserting_transitions.begin(), x_s.q->outgoing_blank_inserting_transitions.end(), v, in, op);
387 #endif
388  }
389 
390  //Process call-transitions; Earley's predictor
391  for (outgoing_call_transition_set_iterator call_transition_begin(x_s.q->outgoing_call_transitions.begin()); call_transition_begin != x_s.q->outgoing_call_transitions.end(); ++call_transition_begin)
392  {
393  //New paused execution state (q_r, b, q_c, q_h, j)
394  std::pair<paused_execution_state_set_const_iterator, bool> x_p_result(v.pess.insert(paused_execution_state(static_cast<state_const_ref>(call_transition_begin->target), static_cast<state_const_ref>(call_transition_begin->called), x_s.q_h, x_s.i)));
395  if (x_p_result.second)
396  {

399 #endif

401  if (t.find(static_cast<state_const_ref>(call_transition_begin->called)) != t.end())
402  {
403  std::pair<active_execution_state_set_const_iterator, bool> x_t_result(v.aess.insert(active_execution_state(static_cast<state_const_ref>(call_transition_begin->target), x_s.q_h, x_s.i)));
404  if (x_t_result.second)
405  {
406 #ifdef TRACE
407  x_t_result.first->serialize(std::wcout) << L" epsilon-completion" << std::endl;
408 #endif

410  op(*x_t_result.first);
411  }
412  }
413  else
414  {

416  std::pair<active_execution_state_set_const_iterator, bool> x_t_result(v.aess.insert(active_execution_state(static_cast<state_const_ref>(call_transition_begin->called), static_cast<state_const_ref>(call_transition_begin->called), idx)));
417  if (x_t_result.second)
418  {
419  e.push(&(*x_t_result.first));

423 #endif
424  }
425  }
426  }
427  }

430  if (x_s.q->is_final())
431  {
432  //Dummy paused execution state used as lower bound for the retrieval of every paused execution state having q_c == q_h
433  dummy_x_p.q_c = x_s.q_h;
434  paused_execution_state_set_iterator pk(the_chart[x_s.i].pess.lower_bound(dummy_x_p));
435 
436  //If this is a completion of a subanalysis, that is, there is at least one superanalysis depending on it (if not, there is nothing to be completed)
437  if (pk != the_chart[x_s.i].pess.end() && pk->q_c == x_s.q_h)
438  {
439  //If this is an epsilon-completion, store epsilon translation for future completions within the same Earley chart item
440  if (x_s.i == idx)
441  t.insert(x_s.q_h);
442 
443  //For each paused execution state (q', q_h, q_h', i) in V_j
444  do
445  {

447  std::pair<active_execution_state_set_const_iterator, bool> result(v.aess.insert(active_execution_state(pk->q, pk->q_h, pk->i)));
448  if (result.second)

452 #endif
453  e.push(&(*result.first));
454  op(*result.first);
455  }
456  ++pk;
457  }while (pk != the_chart[x_s.i].pess.end() && pk->q_c == x_s.q_h);
458  }
459  }
460  }

529  //Compute the r-translations of the input range [input_being, input_end) and add them to the set t
530  bool operator()(const machine& grammar, SourceRef input_begin, SourceRef input_end, bool hasnt_white_at_begin, bool hasnt_white_at_end)
531  {
532 #ifdef TRACE
533  std::wcout << L"Begin parsing\n";
534 #endif
535  //Initialize chart
536  std::size_t chart_size(std::distance(input_begin, input_end) + 1);
537  the_chart.reset(chart_size);
538 
539  bool accept(false);
540  no_op_on_execution_state no_op;
541  acceptor_execution_state_acknowledger ack_op(accept, grammar.initial_state());
542 
543  //Build V_0 as the closure of {(q, epsilon, lambda, q, 0) | q in Q_I};
544  //If empty input, build initial and final V[0]
545  if (input_begin == input_end)
546  {
547  build_initial_ses(grammar.initial_state(), hasnt_white_at_begin, ack_op);
548  //First token is white separated if there are trailing whites at the beginning
549  eclosure(0, input_begin, hasnt_white_at_begin, ack_op);
550  }
551  //Else build initial V[0] and the remaining V[i]
552  else
553  {
554  build_initial_ses(grammar.initial_state(), hasnt_white_at_begin, no_op);
555  //First token is white separated if there are trailing whites at the beginning
556  eclosure(0, input_begin, hasnt_white_at_begin, no_op);
557 
558  std::size_t idx(0);
559  //While the last chart item active set of execution states is not empty and there are input symbols left, compute the next chart item
560  std::size_t last_idx(chart_size - 2);
561  SourceRef input_begin_plus_1(input_begin);
562  while (!the_chart[idx].aess.empty() && idx < last_idx)
563  {
564 #ifdef TRACE
565  std::wcout << "----- V[" << static_cast<int>(idx) + 1 << "] -----" << std::endl;
566 #endif
567  ++input_begin_plus_1;
568  translate_symbol(idx, input_begin, no_op);
569  ++idx;
570  eclosure(idx, input_begin_plus_1, (*input_begin)->end == (*input_begin_plus_1)->begin, no_op);
571  ++input_begin;
572  }
573 
574  //Compute last chart item
575  if (!the_chart[idx].aess.empty())
576  {
577 #ifdef TRACE
578  std::wcout << "----- V[" << static_cast<int>(idx) + 1 << "] -----" << std::endl;
579 #endif
580  ++input_begin_plus_1;
581  translate_symbol(idx, input_begin, ack_op);
582  ++idx;
583  //Next token is the input end... it is white separated if there are trailing whites at the end
584  eclosure(idx, input_begin_plus_1, hasnt_white_at_end, ack_op);
585  ++input_begin;
586  }
587  }
588  the_chart.reset();
589 #ifdef TRACE
590  std::wcout << L"End parsing\n";
591 #endif
592  return accept;
593  }
