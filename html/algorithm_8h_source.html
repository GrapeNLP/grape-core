38  //Performs lexicographical comparison between two ranges r1 and r2 and returns -1 if r1 < r2, 0 if r1 == r2 and 1 if r1 > r2
39  template <typename InputIterator1, typename InputIterator2, typename Compare3W = compare_3w<typename std::iterator_traits<InputIterator1>::value_type> >
40  struct lexicographical_comparator_3way
41  {
42  std::ptrdiff_t operator() (InputIterator1 first1, InputIterator1 last1, InputIterator2 first2, InputIterator2 last2)
43  {
44  std::ptrdiff_t result;
45  while (first1 != last1 && first2 != last2)
46  {
47  if ((result = Compare3W()(*first1, *first2)))
48  return result;
49  ++first1;
50  ++first2;
51  }
52  return static_cast<std::ptrdiff_t>(first2 == last2) - static_cast<std::ptrdiff_t>(first1 == last1);
53  }
54  };

243  //Computes if a sequence has another sequence as suffix;
244  //Comparisons are computed in reverse order, thus end iterators must accept the operator --
245  //The end iterator of the first sequence is passed by reference so after the call
246  //it will point to the last compared element (the first element to differ
247  //or the element corresponding to the first element of the suffix)
248  template<typename Iterator1, typename Iterator2>
249  bool ends_with(Iterator1 lemma_begin, Iterator1 lemma_end, Iterator2 suffix_begin, Iterator2 suffix_end)
250  {
251  //If empty suffix, return true (every string has an empty suffix)
252  if (suffix_begin == suffix_end)
253  return true;
254  //If empty lemma but non-empty suffix, return false (empty strings contain no other suffix than the empty suffix)
255  if (lemma_begin == lemma_end)
256  return false;
257  --lemma_end;
258  --suffix_end;
259  //Compare non-empty lemma and suffix
260  while (*lemma_end == *suffix_end && suffix_end != suffix_begin && lemma_end != lemma_begin)
261  {
262  --lemma_end;
263  --suffix_end;
264  }
265  //Return true iff last compared chars match and every suffix char has been matched
266  return *lemma_end == *suffix_end && suffix_end == suffix_begin;
267  }

300  //Appends to a container a escaped copy of a sequence, that is, every special char will be preceded by the escape char
301  template<typename InputIterator, typename OutputContainer, typename SpecialCharIterator>
302  void escape_special_chars(InputIterator input_begin, InputIterator input_end, OutputContainer &output_container, SpecialCharIterator special_char_begin, SpecialCharIterator special_char_end, typename std::iterator_traits<InputIterator>::value_type escape_char)
303  {
304  //Reserve at least twice the size of the original input sequence: the worst case resulting size
305  output_container.reserve(2*std::distance(input_begin, input_end));
306  for (; input_begin != input_end; ++input_begin)
307  {
308  //If the current char is special, append escape char
309  if (contains(special_char_begin, special_char_end, *input_begin))
310  output_container.pushback(escape_char);
311  //Append current char
312  output_container.pushback(*input_begin);
313  }
314  //Free the unused space
315  output_container.reserve(output_container.size());
316  }
