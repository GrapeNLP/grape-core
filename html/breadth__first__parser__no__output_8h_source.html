235  //Increment parsing chart with the epsilon-closure of the last active set of execution states
236  template<typename ExtraInsertOp>
237  void eclosure(SourceRef in, bool next_token_isnt_white_separated, ExtraInsertOp op)
238  {
239  //While there are execution states pending to be processed
240  while (!e.empty())
241  {
242  //Pop next pending active execution state (q, b, lambda, q_h, j)
243  const execution_state &x_s(*e.front());
244  e.pop();
245 
246  //Process epsilon and inserting transitions
247  process_epsilon_transitions(x_s, x_s.q->outgoing_epsilon_transitions.begin(), x_s.q->outgoing_epsilon_transitions.end(), op);
248  process_inserting_transitions(x_s, x_s.q->outgoing_inserting_transitions.begin(), x_s.q->outgoing_inserting_transitions.end(), in, op);
249 
250  //Process no-blank epsilon and inserting transitions
251  //if there are no whites between the current (or input begin) and the next token (or input end)
252  if (next_token_isnt_white_separated)
253  {
254 #ifdef TRACE
255  process_epsilon_transitions(x_s, x_s.q->outgoing_no_blank_epsilon_transitions.begin(), x_s.q->outgoing_no_blank_epsilon_transitions.end(), op, L"no-blank ");
256  process_inserting_transitions(x_s, x_s.q->outgoing_no_blank_inserting_transitions.begin(), x_s.q->outgoing_no_blank_inserting_transitions.end(), in, op, L"no-blank ");
257 #else
258  process_epsilon_transitions(x_s, x_s.q->outgoing_no_blank_epsilon_transitions.begin(), x_s.q->outgoing_no_blank_epsilon_transitions.end(), op);
259  process_inserting_transitions(x_s, x_s.q->outgoing_no_blank_inserting_transitions.begin(), x_s.q->outgoing_no_blank_inserting_transitions.end(), in, op);
260 #endif
261  }
262 
263  //Process blank epsilon and inserting transitions otherwise
264  else
265  {
266 #ifdef TRACE
267  process_epsilon_transitions(x_s, x_s.q->outgoing_blank_epsilon_transitions.begin(), x_s.q->outgoing_blank_epsilon_transitions.end(), op, L"blank ");
268  process_inserting_transitions(x_s, x_s.q->outgoing_blank_inserting_transitions.begin(), x_s.q->outgoing_blank_inserting_transitions.end(), in, op, L"blank ");
269 #else
270  process_epsilon_transitions(x_s, x_s.q->outgoing_blank_epsilon_transitions.begin(), x_s.q->outgoing_blank_epsilon_transitions.end(), op);
271  process_inserting_transitions(x_s, x_s.q->outgoing_blank_inserting_transitions.begin(), x_s.q->outgoing_blank_inserting_transitions.end(), in, op);
272 #endif
273  }
274 
275  //Process call-transitions; Earley's predictor
276  for (outgoing_call_transition_set_iterator call_transition_begin(x_s.q->outgoing_call_transitions.begin()); call_transition_begin != x_s.q->outgoing_call_transitions.end(); ++call_transition_begin)
277  {
278  state_const_ref target(static_cast<state_const_ref>(call_transition_begin->target));
279  std::pair<execution_state_set_iterator, bool> result(v.insert(execution_state(static_cast<state_const_ref>(call_transition_begin->called), x_s.pi + target)));
280  if (result.second)
281  {
282 #ifdef TRACE
283  result.first->serialize(std::wcout) << L" push" << std::endl;
284 #endif
285  e.push(&(*result.first));
286  op(*result.first);
287  }
288  }
289 
290  //Process pop-transition; Earley's completer
291  if (x_s.q->is_final() && !x_s.pi.empty())
292  {
293  std::pair<execution_state_set_iterator, bool> result(v.insert(execution_state(static_cast<state_const_ref>(x_s.pi.back()), x_s.pi.prefix())));
294  if (result.second)
295  {
296 #ifdef TRACE
297  result.first->serialize(std::wcout) << L" pop" << std::endl;
298 #endif
299  e.push(&(*result.first));
300  op(*result.first);
301  }
302  }
303  }
304  }

307  //Earley scanner: adds a new item to the parsing chart and initializes it with the attainable set of execution states from the former last chart item set of execution states through the consumption of an input symbol sigma
308  template<typename ExtraInsertOp>
309  execution_state_set translate_symbol(SourceRef in_i, ExtraInsertOp op)
310  {
311  execution_state_set w;
312 
313  //For each (q, b, lambda, q_h, i) in v
314  for (execution_state_set_const_iterator x_s_it(v.begin()); x_s_it != v.end(); ++x_s_it)
315  {
316  //For each (q', lambda) = delta(q, (sigma, epsilon))
317  //Process deleting transitions
318  for (outgoing_deleting_transition_set_const_iterator dt_it(x_s_it->q->outgoing_deleting_transitions.begin()); dt_it != x_s_it->q->outgoing_deleting_transitions.end(); ++dt_it)
319  {
320  if (input_match(*in_i, dt_it->input))
321  {
322  //w = w union (q', b, lambda, q_h, i)
323  std::pair<execution_state_set_iterator, bool> result(w.insert(execution_state(static_cast<state_const_ref>(dt_it->target), x_s_it->pi)));
324  if (result.second)
325  {
326 #ifdef TRACE
327  result.first->serialize(std::wcout) << L" (";
328  input_serializer()(std::wcout, *in_i) << L" : <E>)" << std::endl;
329 #endif
330  e.push(&(*result.first));
331  op(*result.first);
332  }
333  }
334  }
335  //For each (q', lambda) = delta(q, (sigma, g))
336  //Process translating transitions
337  for (outgoing_translating_transition_set_const_iterator tt_it(x_s_it->q->outgoing_translating_transitions.begin()); tt_it != x_s_it->q->outgoing_translating_transitions.end(); ++tt_it)
338  {
339  if (input_match(*in_i, tt_it->input))
340  {
341  //w = w union (q', bg, lambda, q_h, i)
342  std::pair<execution_state_set_iterator, bool> result(w.insert(execution_state(static_cast<state_const_ref>(tt_it->target), x_s_it->pi)));
343  if (result.second)
344  {
345 #ifdef TRACE
346  result.first->serialize(std::wcout) << L" (";
347  input_serializer()(std::wcout, *in_i) << L" : ";
348  tag_output_serializer()(std::wcout, tt_it->output) << L')' << std::endl;
349 #endif
350  e.push(&(*result.first));
351  op(*result.first);
352  }
353  }
354  }
355  }
356  return w;
357  }

375  //Compute the r-translations of the input range [input_being, input_end) and add them to the set t
376  bool operator()(const machine& grammar, SourceRef input_begin, SourceRef input_end, bool hasnt_white_at_begin, bool hasnt_white_at_end)
377  {
378 #ifdef TRACE
379  std::wcout << L"Begin parsing\n";
380 #endif
381  bool accept(false);
382 
383  //Initialize pool of state sequences
384  state_const_ref_stack_pool scrsp;
385 
386  //Initialize sets of execution states V and V^prime
387  v.clear();
388 
389  no_op_on_execution_state no_op;
390  acceptor_execution_state_acknowledger ack_op(accept);
391 
392  //Build V_0 as the closure of Q X {b_emptyset} X {lambda};
393  //If empty input, build initial and final V[0]
394  if (input_begin == input_end)
395  {
396  build_initial_ses(grammar.initial_state(), scrsp.empty(), hasnt_white_at_begin, ack_op);
397  //First token is white separated if there are trailing whites at the beginning
398  eclosure(input_begin, hasnt_white_at_begin, ack_op);
399  }
400  //Else build initial V[0] and the remaining V[i]
401  else
402  {
403  build_initial_ses(grammar.initial_state(), scrsp.empty(), hasnt_white_at_begin, no_op);
404  //First token is white separated if there are trailing whites at the beginning
405  eclosure(input_begin, hasnt_white_at_begin, no_op);
406 #ifdef TRACE
407  std::size_t idx(0);
408 #endif
409  SourceRef input_begin_plus_1(input_begin);
410  --input_end;
411  //While the last set of execution states is not empty and there are input symbols left, compute the next set of execution states
412  while (!v.empty() && input_begin != input_end)
413  {
414 #ifdef TRACE
415  std::wcout << L"----- V[" << static_cast<int>(idx) + 1 << L"] -----" << std::endl;
416 #endif
417  ++input_begin_plus_1;
418  execution_state_set w(translate_symbol(input_begin, no_op));
419 #ifdef TRACE
420  ++idx;
421 #endif
422  v.swap(w);
423  eclosure(input_begin_plus_1, (*input_begin)->end == (*input_begin_plus_1)->begin, no_op);
424  ++input_begin;
425  }
426  ++input_end;
427 
428  //Compute last set of execution states
429  if (!v.empty())
430  {
431 #ifdef TRACE
432  std::wcout << L"Last----- V[" << static_cast<int>(idx) + 1 << L"] -----" << std::endl;
433 #endif
434  ++input_begin_plus_1;
435  execution_state_set w(translate_symbol(input_begin, ack_op));
436 #ifdef TRACE
437  ++idx;
438 #endif
439  v.swap(w);
440  //Next token is the input end... it is white separated if there are trailing whites at the end
441  eclosure(input_begin_plus_1, hasnt_white_at_end, ack_op);
442  ++input_begin;
443  }
444  }
445 #ifdef TRACE
446  std::wcout << L"End parsing\n";
447 #endif
448  return accept;
449  }
