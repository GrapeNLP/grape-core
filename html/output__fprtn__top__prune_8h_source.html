141  //Remove all paths which do not reach the specified state from anywhere.
142  //Notice that after parsing, every state in the OUTPUT_FPRTN is derivable from an initial state except

154  //the OUTPUT_FPRTN is equivalent to an empty OUTPUT_FPRTN, so clear it and return
155  if (m.global_acceptor_state.second.incoming_filtered_pop_transitions.empty())
156  {
