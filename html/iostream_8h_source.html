57  virtual int_type overflow(int_type c)
58  {
59  int_type const eof = traits::eof();
60

62  return traits::not_eof(c);
63  else
64  {
65  char_type const ch = traits::to_char_type(c);
66  int_type const r1 = sb1->sputc(ch);
67  int_type const r2 = sb2->sputc(ch);
68  return traits::eq_int_type(r1, eof) || traits::eq_int_type(r2, eof) ? eof : c;
