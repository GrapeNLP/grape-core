999  void erase(base_node_ref z)
1000  {
1001  base_node_ref y(has_missing_child(z) ? z : z->next_ref);
1002  base_node_ref x(has_left(y) ? y->left_ref : y->right_ref);
1003 
1004  x->parent_ref = y->parent_ref;
1005  if (is_root(y))
1006  root_ref() = x;
1007  else if (is_left_child(y))
1008  y->parent_ref->left_ref = x;
1009  else y->parent_ref->right_ref = x;
1010 
1011  bool y_was_red(y->is_red);
1012  if (y != z)
1013  {
1014 // Instead of exchanging the keys, exchange the pointers in order to really erase z
1015 // This way iterators to y will not be invalidated, though this operation might be
1016 // more expensive if the data is smaller than de set of pointers to modify
1017 // As well, this allows the keys to be constant
1018 // z->key() = y->key();
1019  y->parent_ref = z->parent_ref;
1020  if (is_left_child(z))
1021  z->parent_ref->left_ref = y;
1022  else z->parent_ref->right_ref = y;
1023  y->left_ref = z->left_ref;
1024  z->left_ref->parent_ref = y;
1025  y->right_ref = z->right_ref;
1026  z->right_ref->parent_ref = y;
1027  y->is_red = z->is_red;
1028  if (is_root(z))
1029  root_ref() = y;
1030  }
1031  if (!y_was_red)
1032  erase_fixup(x);
1033  --d.node_count;
1034  unbalanced_erase(z);
1035  }
