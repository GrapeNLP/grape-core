212 #endif
213  }
214  //Remove the incoming consuming transition corresponding to the current outgoing consuming transition
215  r_t->second.incoming_consuming_transitions.erase(r_s->second.outgoing_consuming_transitions.front().r_trans_it);
216  //Remove the current outgoing consuming transition
217  r_s->second.outgoing_consuming_transitions.pop_front();
218  if (r_t->second.incoming_consuming_transitions.empty() && r_t->second.incoming_epsilon_transitions.empty() && r_t->second.incoming_call_transitions.empty() && !r_t->second.ifptc)

224  }

232

235  std::wcout << L"processing outgoing call transition (" << r_s->second.wlabel() << L", " << r_s->second.outgoing_call_transitions.front().called->second.wlabel() << L", " << r_t->second.wlabel() << L')' << std::endl;

296  blackboard operator()(machine& m, const blackboard &empty_blackboard = blackboard(the_weight_transformer_traits::identity()))
297  {
298  typedef std::deque<state_const_ref> state_const_ref_stack;
299 #ifdef TRACE
300  typedef typename state_const_ref_stack::const_iterator state_const_ref_stack_const_iterator;
301 #endif
302 
303  top_reverse_path(m, the_weight_transformer_traits::identity());
304 #ifdef TRACE
305  std::wcout << L"Begin top blackboard construction\n";
306 #endif
307  state_const_ref r_t(&m.global_acceptor_state);
308  blackboard b_t(empty_blackboard);
309  state_const_ref_stack pi;
310 #ifdef TRACE
311  b_t.w = m.global_acceptor_state.second.top_w;
312  std::wcout << '(' << r_t->second.wlabel() << ", ";
313  blackboard_serializer()(std::wcout, b_t) << ", ";
314  for (state_const_ref_stack_const_iterator i(pi.begin()); i != pi.end(); ++i)
315  std::wcout << (*i)->second.wlabel();
316  std::wcout << ')' << std::endl;
317 #endif
318  do
319  {
320  switch (r_t->second.top_reverse_transition_type)
321  {
322  case OUTPUT_FPRTN_REVERSE_POP_TRANSITION_TYPE:
323  r_t = pi.back();
324  pi.pop_back();
325  break;
326  case OUTPUT_FPRTN_REVERSE_PUSH_TRANSITION_TYPE:
327  pi.push_back(static_cast<state_ref>(r_t->second.top_r_t));
328  r_t = static_cast<state_const_ref>(r_t->second.top_r_c);
329  break;
330  case OUTPUT_FPRTN_REVERSE_EPSILON_TRANSITION_TYPE:
331  r_t = static_cast<state_ref>(r_t->second.top_r_t);
332  break;
333  case OUTPUT_FPRTN_REVERSE_CONSUMING_TRANSITION_TYPE:
334  cgamma(b_t, r_t->second.src_ref, r_t->second.top_non_w);
335  r_t = static_cast<state_ref>(r_t->second.top_r_t);
336  break;
337  }
338 #ifdef TRACE
339  b_t.w = r_t->second.top_w;
340  std::wcout << '(' << r_t->second.wlabel() << ", ";
341  blackboard_serializer()(std::wcout, b_t) << ", ";
342  for (state_const_ref_stack_const_iterator i(pi.begin()); i != pi.end(); ++i)
343  std::wcout << (*i)->second.wlabel();
344  std::wcout << ')' << std::endl;
345 #endif
346  } while (r_t != &m.initial_state);
347 #ifndef TRACE
348  b_t.w = m.global_acceptor_state.second.top_w;
349 #else
350  std::wcout << L"End of top blackboard construction\n";
351 #endif
352  return b_t;
353  }
