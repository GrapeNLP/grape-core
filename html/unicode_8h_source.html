319 template<typename IntType>
320 unichar u_fread_optional_int(FILE *f, IntType &n, IntType default_value = 0)
321 {
322  unichar c(u_fgetc(f));
323  if (feof(f))
324  {
325  n = default_value;
326  return c;
327  }
328  IntType sign(1);
329  if (c == '-')
330  {
331  sign = -1;
332  c = fgetc(f);
333  }
334  else if (c == '+')
335  c = fgetc(f);
336  else if (!u_is_digit(c))
337  {
338  n = default_value;
339  return c;
340  }
341  n = 0;
342  do
343  {
344  n = n * ((IntType)10) + ((IntType)c) - ((IntType)'0');
345  c = u_fgetc(f);
346  }while (!feof(f) && u_is_digit(c));
347  n *= sign;
348  return c;
349 }

412 template<typename FloatType>
413 unichar u_fread_float(FILE *f, FloatType &n, unichar c)
414 {
415  std::string s;
416  //Read sign
417  if (c == '-' || c == '+')
418  {
419  s += static_cast<char>(c);
420  c = u_fgetc(f);
421  }
422  //Catch special case "Not a Number"
423  if ((c == 'n' || c == 'N'))
424  {
425  c = u_fgetc(f);
426  if (c != 'a' && c != 'A')
427  fatal_error("Expected 'a' or 'A' while reading \"Not a Number\" but got %c", c);
428  c = u_fgetc(f);
429  if (c != 'n' && c != 'N')
430  fatal_error("Expected 'n' or 'N' while reading \"Not a Number\" but got %c", c);
431  if (s == "-")
432  n = -std::numeric_limits<FloatType>::quiet_NaN();
433  else n = std::numeric_limits<FloatType>::quiet_NaN();
434  return u_fgetc(f);
435  }
436  //Catch special case "Infinity"
437  if ((c == 'i' || c == 'I'))
438  {
439  c = u_fgetc(f);
440  if (c != 'n' && c != 'N')
441  fatal_error("Expected 'n' or 'N' while reading \"Infinity\" but got %c", c);
442  c = u_fgetc(f);
443  if (c != 'f' && c != 'F')
444  fatal_error("Expected 'f' or 'F' while reading \"Infinity\" but got %c", c);
445  if (s == "-")
446  n = -std::numeric_limits<FloatType>::infinity();
447  else n = std::numeric_limits<FloatType>::infinity();
448  return u_fgetc(f);
449  }
450  //Read mantissa's integer part
451  while (!feof(f) && u_is_digit(c))
452  {
453  s += static_cast<char>(c);
454  c = u_fgetc(f);
455  }
456  //Read mantissa's decimal part
457  if (!feof(f) && c == '.')
458  {
459  //Add dot
460  s += static_cast<char>(c);
461  c = u_fgetc(f);
462  //Read decimal digits
463  while (!feof(f) && u_is_digit(c))
464  {
465  s += static_cast<char>(c);
466  c = u_fgetc(f);
467  }
468  }
469  //Read exponent
470  if (!feof(f) && (c == 'E' || c == 'e'))
471  {
472  s += static_cast<char>(c);
473  c = u_fgetc(f);
474  //Read sign
475  if (c == '-' || c == '+')
476  {
477  s += static_cast<char>(c);
478  c = u_fgetc(f);
479  }
480  //Read digits
481  while (!feof(f) && u_is_digit(c))
482  {
483  s += static_cast<char>(c);
484  c = u_fgetc(f);
485  }
486  }
487  n = boost::lexical_cast<FloatType>(s);
488  return c;
489 }
