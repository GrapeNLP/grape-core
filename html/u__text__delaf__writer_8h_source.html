44  void u_write(FILE *f, const u_text_delaf<CaseNormalizer> &dico)
45  {
46  typename u_text_delaf<CaseNormalizer>::u_lemma_trie_with_entry_data::const_iterator i(dico.lemma_to_entry_index.begin());
47 
48  //While there are lemma trie nodes to process
49  while (i != dico.lemma_to_entry_index.end())
50  {
51  //If this node represents a complete word, write it
52  //Otherwise, just skip it
