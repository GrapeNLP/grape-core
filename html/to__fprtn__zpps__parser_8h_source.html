519  //Compute the r-translations of the input range [input_being, input_end) and add them to the set t
520  void operator()(const in_machine& grammar, bool hasnt_white_at_begin, bool hasnt_white_at_end, out_machine& out)
521  {
522 #ifdef TRACE
523  std::wcout << "Begin parsing\n";
524  //Keep the original input begin for drawing the resulting FPRTN
525 #endif
526  SourceRef input_begin(out.initial_state.second.src_ref);
527  SourceRef input_end(out.global_acceptor_state.second.src_ref);
528 
529  //Add common paused execution state for every initial call
530  //The only important data here is the FPRTN initial state,
531  //which will be used for computing the composition of its initial
532  //blackboard set {b_empty} with the blackboard of each FPRTN state
533  //accepting an initial call
534  std::pair<paused_execution_state_map_iterator, bool> result(out.the_chart[0].pess.insert(paused_execution_state(NULL, &out.initial_state, NULL, NULL, 0)));
535 #ifdef TRACE
536  std::wcout << "added init pes for init state " << out.initial_state.second.wlabel() << std::endl;
537 #endif
538  no_op_on_execution_state no_op;
539  acceptor_execution_state_inserter ins_op(out.global_acceptor_state, grammar.initial_state(), result.first);
540 
541  //Build V_0 as the closure of {(q, {r_I}, lambda, q, 0) | q in Q_I};
542  //If empty input, build initial and final V[0]
543  if (input_begin == input_end)
544  {
545  build_initial_ses(grammar.initial_state(), hasnt_white_at_begin, out, ins_op);
546  //First token is white separated if there are trailing whites at the beginning
547  eclosure(0, input_begin, hasnt_white_at_begin, out, ins_op);
548  }
549  //Else build initial V[0] and the remaining V[i]
550  else
551  {
552  build_initial_ses(grammar.initial_state(), hasnt_white_at_begin, out, no_op);
553  //First token is white separated if there are trailing whites at the beginning
554  eclosure(0, input_begin, hasnt_white_at_begin, out, no_op);
555 
556  std::size_t idx(0);
557  //While the last chart item active set of execution states is not empty and there are input symbols left, compute the next chart item
558  std::size_t last_idx(out.the_chart.size() - 2);
559  SourceRef input_begin_plus_1(input_begin);
560  while (!out.the_chart[idx].other_states.empty() && idx < last_idx)
561  {
562 #ifdef TRACE
563  std::wcout << "----- V[" << static_cast<int>(idx) + 1 << "] -----" << std::endl;
564 #endif
565  ++input_begin_plus_1;
566  translate_symbol(idx, input_begin, input_begin_plus_1, out, no_op);
567  ++idx;
568  eclosure(idx, input_begin_plus_1, (*input_begin)->end == (*input_begin_plus_1)->begin, out, no_op);
569  ++input_begin;
570  }
571 
572  //Compute last chart item
573  if (!out.the_chart[idx].other_states.empty())
574  {
575 #ifdef TRACE
576  std::wcout << "----- V[" << static_cast<int>(idx) + 1 << "] -----" << std::endl;
577 #endif
578  ++input_begin_plus_1;
579  translate_symbol(idx, input_begin, input_begin_plus_1, out, ins_op);
580  ++idx;
581  //Next token is the input end... it is white separated if there are trailing whites at the end
582  eclosure(idx, input_begin_plus_1, hasnt_white_at_end, out, ins_op);
583  ++input_begin;
584  }
585  }
586 #ifdef TRACE
587  std::wcout << "***** Paused ESs *****\n";
588  std::wcout << "----- V[0] -----\n";
589  paused_execution_state_map_const_iterator x_p_it(out.the_chart[0].pess.begin());
590  std::wcout << "initial paused execution state with source FPRTN state ";
591  x_p_it->first.r->serialize_fprtn_state(std::wcout) << std::endl;
592  for (++x_p_it; x_p_it != out.the_chart[0].pess.end(); ++x_p_it)
593  static_cast<const paused_execution_state&>(*x_p_it).serialize(std::wcout) << std::endl;
594  input_begin = out.initial_state.second.src_ref;
595  std::size_t idx(1);
596  for (; input_begin != input_end; ++input_begin, ++idx)
597  {
598  std::wcout << "----- V[" << static_cast<int>(idx) << "] -----\n";
599  for (x_p_it = out.the_chart[idx].pess.begin(); x_p_it != out.the_chart[idx].pess.end(); ++x_p_it)
600  static_cast<const paused_execution_state&>(*x_p_it).serialize(std::wcout) << std::endl;
601  }
602  std::wcout << "***** End of paused ESs *****\n";
603  std::wcout << "End parsing\n";
604  write_fprtn_dot("dirty_parses", "../dirty_parses.dot", out);
605 #endif
606  }
