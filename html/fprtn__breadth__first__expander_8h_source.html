106  bool operator< (const execution_state& x2) const
107  {
108  if (q < x2.q)
109  return true;
110  if (x2.q < q)
111  return false;
112  //We don't care here about lexicographical comparison, we simply compare the object addresses in order to optimize comparison

279  //Develop first execution state of the queue of pending execution states
280  void traverse(blackboard_set &result)
281  {
282 // std::wcout << L"traverse\n";
283  //If this execution state is complete (it corresponds to a final state with no pending calls)
284  //add it to the result set
285  if (e.front().q->is_final() && state_const_ref_stack_deref()(e.front().pi).empty())
286  {
287  result.insert(e.front().b);
288 #ifdef TRACE
289  e.front().serialize(std::wcout) << L" added to result" << std::endl;
290 #endif
291  }
292  //If its current state has at least one consuming transition, develop them
293  if (!e.front().q->second.outgoing_consuming_transitions.empty())
294  traverse_from_consuming();
295  //If its current state has at least one epsilon transition, develop them
296  else if (!e.front().q->second.outgoing_epsilon_transitions.empty())
297  traverse_from_epsilon();
298  //If its current state has at least one call transition, develop them
299  else if (!e.front().q->second.outgoing_call_transitions.empty())
300  traverse_from_call();
301  //If it is possible to return from a call (its state is final and the return state stack
302  //is not empty), develop the return from call
303  else if (e.front().q->is_final() && !state_const_ref_stack_deref()(e.front().pi).empty() && e.front().q->second.src_ref == state_const_ref_stack_deref()(e.front().pi).back()->second.src_ref)
304  traverse_from_pop();
305  //Otherwise current execution state admits no more development: delete it
306  else
307  {
308 #ifdef TRACE
309  e.front().serialize(std::wcout) << L" deleted" << std::endl;
310 #endif
311  e.pop();
312  }
313  }

315  public:
316  //Computes a set of maps for every complete path within the FPRTN, where the map keys are input segment identifiers
317  //and the map values are pairs of input positions (that is, they mark segments of the input containing relevant
318  //information to be extracted (a phone number) with the identifier of that information (phone)).
319  //We pass the FPRTN to expand, the input begin and end iterators and an empty set of segment maps to be filled
320  //The algorithm represents execution states as triplets (current state, current segment map, return state stack)
321  //It uses a queue to store the execution states that are still to be developed
322  //It first adds to the queue the initial set of execution states as (initial state, empty segment map, empty return state stack)
323  //It derives new execution states from the first execution state of the queue, and pushes them onto the queue
324  //The last derived one is not pushed, but replaces the first execution state (the first one is modified)
325  //If the first execution state has no possible derivations, it is popped out and the process is repeated for the next execution state
326  //until there are no execution states left
327  //Each time an acceptor execution state is reached, its segment map is added to the result
328  //Acceptor execution states have an acceptor state and no pending returns from call (an empty stack of return states)
329  blackboard_set& operator() (const machine& m, blackboard_set &result, const blackboard &empty_blackboard = blackboard())
330  {
331 #ifdef TRACE
332  std::wcout << L"Begin breadth-first expansion\n";
333 #endif
334  //Initialize pool of state sequences
335  state_const_ref_stack_ref_pool scrsp;
336 
337  //For the initial FPRTN state, build an expansion execution state (q, b, lambda) and add it to the expansion queue
338  e.push(execution_state(&m.initial_state, empty_blackboard, scrsp.empty()));
339 #ifdef TRACE
340  e.back().serialize(std::wcout) << L" initial" << std::endl;
341 #endif
342 
343  //While there are pending execution states to develop, pop first execution state from
344  //the queue and develop it
345  while (!e.empty())
346  traverse(result);
347 #ifdef TRACE
348  std::wcout << L"End of breadth-first expansion\n";
349 #endif
350  return result;
351  }
352  };
