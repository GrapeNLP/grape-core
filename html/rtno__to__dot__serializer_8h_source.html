71  std::basic_ostream<CharT, Traits>& serialize_state(std::basic_ostream<CharT, Traits>& out, CharT state_letter, state_const_ref s)
72  {
73  //Set double circle if final state
74  if (s->is_final())
75  {
76  serialize_state_label(out, state_letter, s);
77  out << "[peripheries = 2];" << std::endl;
78  }
79  //Serialize at least the state label when it has no outgoing transitions
80  else if (s->has_no_transitions())
81  return serialize_state_label(out, state_letter, s) << ';' << std::endl;
82 
83  if (s->has_no_transitions())
84  {
85  serialize_state_label(out, state_letter, s) << ';' << std::endl;
86  return out;
87  }
88  //Print epsilon transitions
89  outgoing_epsilon_transition_set_const_iterator oet_it;
90  for (oet_it = s->outgoing_epsilon_transitions.begin(); oet_it != s->outgoing_epsilon_transitions.end(); ++oet_it)
91  {
92  serialize_state_label(out, state_letter, s);
93  out << " -> ";
94  serialize_state_label(out, state_letter, static_cast<state_const_ref>(oet_it->target));
95  out << " [label = \"<E>\"];" << std::endl;
96  }
97  outgoing_inserting_transition_set_const_iterator oit_it;
98  for (oit_it = s->outgoing_inserting_transitions.begin(); oit_it != s->outgoing_inserting_transitions.end(); ++oit_it)
99  {
100  serialize_state_label(out, state_letter, s);
101  out << " -> ";
102  serialize_state_label(out, state_letter, static_cast<state_const_ref>(oit_it->target));
103  out << " [label = \"<E>/";
104  tos(out, oit_it->output) << "\"];" << std::endl;
105  }
106  for (oet_it = s->outgoing_no_blank_epsilon_transitions.begin(); oet_it != s->outgoing_no_blank_epsilon_transitions.end(); ++oet_it)
107  {
108  serialize_state_label(out, state_letter, s);
109  out << " -> ";
110  serialize_state_label(out, state_letter, static_cast<state_const_ref>(oet_it->target));
111  out << " [label = \"#\"];" << std::endl;
112  }
113  for (oit_it = s->outgoing_no_blank_inserting_transitions.begin(); oit_it != s->outgoing_no_blank_inserting_transitions.end(); ++oit_it)
114  {
115  serialize_state_label(out, state_letter, s);
116  out << " -> ";
117  serialize_state_label(out, state_letter, static_cast<state_const_ref>(oit_it->target));
118  out << " [label = \"#/";
119  tos(out, oit_it->output) << "\"];" << std::endl;
120  }
121  for (oet_it = s->outgoing_blank_epsilon_transitions.begin(); oet_it != s->outgoing_blank_epsilon_transitions.end(); ++oet_it)
122  {
123  serialize_state_label(out, state_letter, s);
124  out << " -> ";
125  serialize_state_label(out, state_letter, static_cast<state_const_ref>(oet_it->target));
126  out << " [label = \"'\\ '\"];" << std::endl;
127  }
128  for (oit_it = s->outgoing_blank_inserting_transitions.begin(); oit_it != s->outgoing_blank_inserting_transitions.end(); ++oit_it)
129  {
130  serialize_state_label(out, state_letter, s);
131  out << " -> ";
132  serialize_state_label(out, state_letter, static_cast<state_const_ref>(oit_it->target));
133  out << " [label = \"'\\ '/";
134  tos(out, oit_it->output) << "\"];" << std::endl;
135  }
136 
137  //Print deleting transitions
138  outgoing_deleting_transition_set_const_iterator odt_it;
139  for (odt_it = s->outgoing_deleting_transitions.begin(); odt_it != s->outgoing_deleting_transitions.end(); ++odt_it)
140  {
141  serialize_state_label(out, state_letter, s);
142  out << " -> ";
143  serialize_state_label(out, state_letter, static_cast<state_const_ref>(odt_it->target));
144  out << " [label = \"";
145  tis(out, odt_it->input) << "\"];" << std::endl;
146  }
147 
148  //Print translating transitions
149  outgoing_translating_transition_set_const_iterator ott_it;

151  {
152  serialize_state_label(out, state_letter, s);
153  out << " -> ";
154  serialize_state_label(out, state_letter, static_cast<state_const_ref>(ott_it->target));
155  tis(out << " [label = \"", ott_it->input) << '/';

158 
159  //Print call transitions
160  outgoing_call_transition_set_const_iterator ocallt_it;

162  {
163  serialize_state_label(out, state_letter, s);
164  out << " -> ";
165  serialize_state_label(out, state_letter, static_cast<state_const_ref>(ocallt_it->target));

168  }
169  return out;

172  template<typename CharT, typename Traits>
173  std::basic_ostream<CharT, Traits>& operator() (std::basic_ostream<CharT, Traits>& out, const char *rtno_name, CharT state_letter, const machine &m)
174  {
175  out << "digraph " << rtno_name << " {" << std::endl;
176  out << "rankdir=LR" << std::endl;
177  out << "ranksep=0" << std::endl;
178  out << "nodesep=.1" << std::endl;
179 // out << "size=\"8,5\"" << std::endl;
180 // out << "node [shape = ellipse];" << std::endl;
181 
182  //Print initial states (states having an arrow coming from nowhere, that is, from an invisible state "ini_i")
183  state_ref_list_const_iterator i(m.states.begin());
184  //Print arrow towards initial state
185  out << "ini" << " [style = invis];" << std::endl;
186  //Print initial transition
187  out << "ini" << " -> ";
188  serialize_state_label(out, state_letter, m.initial_state());
189  out << ';' << std::endl;
190  //Print states
191  for (state_ref_list_const_iterator i(m.states.begin()); i != m.states.end(); ++i)
192  serialize_state(out, state_letter, static_cast<state_const_ref>(*i));
193  return out << '}' << std::endl;
194  }
