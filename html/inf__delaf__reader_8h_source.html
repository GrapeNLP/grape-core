62  void u_read(FILE *f, inf_delaf &dico)
63  {
64  //Get entry count
65  std::size_t ambiguous_entry_count;
66 // std::wcout << "Reading ambiguous entry count: ";
67  u_fread_unsigned_int(f, ambiguous_entry_count);
68 // std::wcout << ambiguous_entry_count << std::endl;
69  //Reserve space for all the entries
70  dico.entries.reserve(ambiguous_entry_count);
71 
72  //For each entry, load it
73  std::size_t i(0);
74  for (; i < ambiguous_entry_count; ++i)
75  {
76  u_read_inf_delaf_ambiguous_entry(f, ambiguous_entry_count, dico);
77 // std::wcout << ambiguous_entry_separator;
78  }

82  {

93  unichar u_read_inf_delaf_entry(FILE *f, std::size_t ambiguous_entry_count, inf_delaf &dico, inf_delaf::ambiguous_entry &ae)
94  {
95  //Load mandatory canonical form
96  //The canonical form is mandatory but may be empty;
97  //In that case, the lemma is also the canonical form
98  unsigned char inflected_form_suffix_length;
99  u_array canonical_form_suffix;
100  unichar next_char = u_read_inf_delaf_canonical_form(f, inflected_form_suffix_length, canonical_form_suffix);
101 
102  //After the canonical form there must be the separator between it and the POS;
103  //otherwise the DELAF entry is incomplete
104  if (feof(f) || next_char != canonical_form_separator)
105  fatal_error("INF DELAF entry %d truncated after canonical form\n", static_cast<int>(ambiguous_entry_count));
106 
107  //Load mandatory part-of-speech, with is processed as a semantic trait
108 // std::wcout << canonical_form_separator;
109  ust_string::ref pos(&dico.ppooss.epsilon());
110  next_char = u_read_inf_delaf_semantic_trait(f, pos);
111  //After the POS we may find an EOL, an EOF, a semantic trait separator or a flexional trait separator
112  //The semantic traits, flexional traits and clitic pronouns are optional
113  //Anyway, the POS should not be empty
114  if (pos == &dico.ppooss.epsilon())
115  fatal_error("DELAF entry %d having an empty POS\n", static_cast<int>(ambiguous_entry_count));
116 // std::wcout << *pos;
117 
118  std::unique_ptr<inf_delaf::entry> e_ref(new inf_delaf::entry(inflected_form_suffix_length, std::move(canonical_form_suffix), *pos));
119 
120  //Load optional semantic traits
121  //After the POS we may find a sequence of zero, one or more semantic traits
122  if (next_char == semantic_traits_separator)
123  next_char = u_read_inf_delaf_semantic_traits(f, ambiguous_entry_count, dico, e_ref->semantic_traits);
124 
125  //Load optional flexional traits
126  //After the sequence of zero, one or more semantic traits, we may find a sequence of zero, one or more flexional traits
127  if (next_char == flexional_traits_separator)
128  {
129  next_char = u_read_inf_delaf_possible_flexional_traits(f, ambiguous_entry_count, dico, e_ref->possible_flexional_traits);
130  //Load clitic pronouns
131  //If there are flexional traits specified, we may found zero, one or more clitic pronouns afterwards
132  if (next_char == clitic_pronouns_separator)
133  next_char = u_read_inf_delaf_clitic_pronouns(f, ambiguous_entry_count, dico, e_ref->clitic_pronouns);
134  }
135 
136  if (next_char == comment_separator)
137  u_fskip_line(f);
138 
139  ae.push_back(e_ref.release());
140  return next_char;
141  }

143  unichar u_read_inf_delaf_canonical_form(FILE *f, unsigned char &inflected_form_suffix_length, u_array &canonical_form_suffix)
144  {
145  unichar next_char(u_fread_optional_unsigned_int<unsigned char>(f, inflected_form_suffix_length));
146  if (next_char != canonical_form_separator) {
147  u_vector canonical_form_suffix_vector;
148  canonical_form_suffix_vector.push_back(next_char);
149  next_char = u_read(f, canonical_form_suffix_vector, canonical_form_separator);
150  canonical_form_suffix.reset(canonical_form_suffix_vector.begin(), canonical_form_suffix_vector.end());
151 // std::wcout << canonical_form_suffix;
152  }
153  return next_char;
154  }

157  {
158  ust_string::ref semantic_trait;
159  unichar next_char;
160  do
161  {
162 // std::wcout << semantic_traits_separator;
163  semantic_trait = &dico.semantic_traits.epsilon();
164  next_char = u_read_inf_delaf_semantic_trait(f, semantic_trait);
165  if (semantic_trait == &dico.semantic_traits.epsilon())
166  fatal_error("DELAF entry %d having an empty semantic trait\n", static_cast<int>(ambiguous_entry_count));
167  semantic_traits.insert(semantic_trait);

169  } while (!feof(f) && next_char == semantic_traits_separator);
