73  template<typename CharT, typename Traits>
74  std::basic_ostream<CharT, Traits>& serialize_state(std::basic_ostream<CharT, Traits>& out, const state &s, std::size_t i)
75  {
76  //Set double circle if final state
77  if (s.is_final())
78  {
79  serialize_state_label(out, s);
80  out << "[peripheries = 2];" << std::endl;
81  }
82  //No need here for special serialization of states without outgoing transitions since they will at least be linked to their key box
83  //Print key box
84  serialize_state_kappa_label(out, s) << " [shape = box, height = 0, width = 0, label = \"" << i << "\"];" << std::endl;
85  serialize_state_label(out, s);
86  out << " -> ";
87  serialize_state_kappa_label(out, s) << " [dir = none, minlen = 0, len = 0, tailport = s, headport = n];" << std::endl;;
88 
89  //Print epsilon transitions
90  for (outgoing_epsilon_transition_set_const_iterator oet_it(s.second.outgoing_epsilon_transitions.begin()); oet_it != s.second.outgoing_epsilon_transitions.end(); ++oet_it)
91  {
92  serialize_state_label(out, s);
93  out << " -> ";
94  serialize_state_label(out, *oet_it->target);
95  out << " [label = \"<E>\"];" << std::endl;
96  }
97 
98  //Print deleting transitions
99  for (outgoing_consuming_transition_set_const_iterator oconst_it(s.second.outgoing_consuming_transitions.begin()); oconst_it != s.second.outgoing_consuming_transitions.end(); ++oconst_it)
100  {
101  serialize_state_label(out, s);
102  out << " -> ";
103  serialize_state_label(out, *oconst_it->target);
104  out << " [label = \"";
105  tis(out, oconst_it->input) << "\"];" << std::endl;
106  }
107 
108  //Print call and push transitions
109  for (outgoing_call_transition_set_const_iterator ocallt_it(s.second.outgoing_call_transitions.begin()); ocallt_it != s.second.outgoing_call_transitions.end(); ++ocallt_it)
110  {
111  //Call
112  serialize_state_label(out, s);
113  out << " -> ";
114  serialize_state_label(out, *ocallt_it->target);
115  out << " [label = \"";
116  ocallt_it->called->second.serialize(out) << "\", style = dashed];" << std::endl;
117 
118  //Push
119  serialize_state_label(out, s);
120  out << " -> ";
121  serialize_state_label(out, *ocallt_it->called);
122  out << " [label = \"";
123  ocallt_it->target->second.serialize(out) << "\", style = dotted];" << std::endl;
124  }
125 
126  //Print pop transitions
127  for (outgoing_filtered_pop_transition_set_const_iterator ofpopt_it(s.second.outgoing_filtered_pop_transitions.begin()); ofpopt_it != s.second.outgoing_filtered_pop_transitions.end(); ++ofpopt_it)
128  {
129  serialize_state_label(out, s);
130  out << " -> ";
131  serialize_state_label(out, *ofpopt_it->target);
132  out << " [style = bold];" << std::endl;
133  }
134 
135  return out;
136  }

139  std::basic_ostream<CharT, Traits>& operator() (std::basic_ostream<CharT, Traits>& out, const char *output_fprtn_name, const machine &m)
140  {
141  out << "digraph " << output_fprtn_name << " {" << std::endl;
142  out << "rankdir = LR" << std::endl;
143  out << "ranksep = 0" << std::endl;
144  out << "nodesep = .1" << std::endl;
145 // out << "size = \"8,5\"" << std::endl;
146 // out << "node [shape = ellipse];" << std::endl;
147 
148  //Print ini_i state
149  out << "ini" << " [style = invis];" << std::endl;
150  //Print initial transition
151  out << "ini" << " -> ";
152  serialize_state_label(out, m.initial_state);
153  out << ';' << std::endl;
154  serialize_state(out, m.initial_state, 0);
155  serialize_state(out, m.global_acceptor_state, m.the_chart.size() - 1);
156 
157  //Print other states
158  chart_const_iterator c_it(m.the_chart.begin());
159  for (std::size_t i(0); i < m.the_chart.size(); ++i, ++c_it)
160  for (state_const_iterator s_it(c_it->other_states.begin()); s_it != c_it->other_states.end(); ++s_it)
161  serialize_state(out, static_cast<const state&>(*s_it), i);
162  return out << '}' << std::endl;
163  }
