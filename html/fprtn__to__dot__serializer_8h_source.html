71  template<typename CharT, typename Traits>
72  std::basic_ostream<CharT, Traits>& serialize_state(std::basic_ostream<CharT, Traits>& out, state_const_ref s, SourceRef src_begin)
73  {
74  //Set double circle if final state
75  if (s->is_final())
76  {
77  serialize_state_label(out, s);
78  out << "[peripheries = 2];" << std::endl;
79  }
80 
81  //Print key box
82  serialize_state_kappa_label(out, s) << " [shape = box, height = 0, width = 0, label = \"" << std::distance(src_begin, s->src_ref) << "\"];" << std::endl;
83  serialize_state_label(out, s);
84  out << " -> ";
85  serialize_state_kappa_label(out, s) << " [dir = none, minlen = 0, len = 0, tailport = s, headport = n];" << std::endl;;
86 
87  //Print epsilon transitions
88  for (outgoing_epsilon_transition_set_const_iterator oet_it(s->outgoing_epsilon_transitions.begin()); oet_it != s->outgoing_epsilon_transitions.end(); ++oet_it)
89  {
90  serialize_state_label(out, s);
91  out << " -> ";
92  serialize_state_label(out, oet_it->target);
93  out << " [label = \"<E>\"];" << std::endl;
94  }
95 
96  //Print deleting transitions
97  for (outgoing_consuming_transition_set_const_iterator oconst_it(s->outgoing_consuming_transitions.begin()); oconst_it != s->outgoing_consuming_transitions.end(); ++oconst_it)
98  {
99  serialize_state_label(out, s);
100  out << " -> ";
101  serialize_state_label(out, oconst_it->target);
102  out << " [label = \"";
103  tis(out, oconst_it->input) << "\"];" << std::endl;
104  }
105 
106  //Print call and push transitions
107  for (outgoing_call_transition_set_const_iterator ocallt_it(s->outgoing_call_transitions.begin()); ocallt_it != s->outgoing_call_transitions.end(); ++ocallt_it)
108  {
109  //Call
110  serialize_state_label(out, s);
111  out << " -> ";
112  serialize_state_label(out, ocallt_it->target);
113  out << " [label = \"";
114  ocallt_it->called->serialize(out) << "\", style = dashed];" << std::endl;
115 
116  //Push
117  serialize_state_label(out, s);
118  out << " -> ";
119  serialize_state_label(out, ocallt_it->called);
120  out << " [label = \"";
121  ocallt_it->target->serialize(out) << "\", style = dotted];" << std::endl;
122  }
123 
124  //Print pop transitions
125  for (outgoing_filtered_pop_transition_set_const_iterator ofpopt_it(s->outgoing_filtered_pop_transitions.begin()); ofpopt_it != s->outgoing_filtered_pop_transitions.end(); ++ofpopt_it)
126  {
127  serialize_state_label(out, s);
128  out << " -> ";
129  serialize_state_label(out, ofpopt_it->target);
130  out << " [style = bold];" << std::endl;
131  }
132 
133  return out;
134  }

147  int ini_count;
148  state_ref_list_const_iterator i;
149  for (i = m.ini_states.begin(), ini_count = 0; i != m.ini_states.end(); ++i, ++ini_count)
150  {
151  //Print ini_i state
152  out << "ini_" << ini_count << " [style = invis];" << std::endl;
153  //Print initial transition
154  out << "ini_" << ini_count << " -> ";
155  serialize_state_label(out, *i);
156  out << ';' << std::endl;
157  serialize_state(out, *i, src_begin);
158  }
159 
160  //Print other states
161  for (i = m.other_states.begin(); i != m.other_states.end(); ++i)
162  serialize_state(out, *i, src_begin);
163  return out << '}' << std::endl;
