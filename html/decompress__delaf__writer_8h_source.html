84  void u_decompress_delaf_write(FILE *f, const compressed_delaf::word::u_const_iterator word_begin, compressed_delaf::word::u_const_iterator word_end, inf_delaf::entry &e)
85  {
86  //Write inflected form
87  u_write<compressed_delaf::word::u_const_iterator>(f, word_begin, word_end);
88  u_fputc(lemma_separator, f);
89  //Write canonical form, if the inflected form is not already the canonical form
90  if (!e.is_canonical_form())
91  u_decompress_delaf_write_canonical_form(f, word_begin, word_end, e.inflected_form_suffix_length, e.canonical_form_suffix.begin(), e.canonical_form_suffix.end());
92  u_fputc(canonical_form_separator, f);
93  //Write Part-Of-Speech
94  u_write(f, e.pos.begin(), e.pos.end());
95  //Write semantic traits, if present
96  if (!e.semantic_traits.empty())
97  {
98  u_fputc(semantic_traits_separator, f);
99  u_ptr_write(f, e.semantic_traits.begin(), e.semantic_traits.end(), semantic_traits_separator);
100  }
101  //Write flexional traits, if present
102  if (!e.possible_flexional_traits.empty())
103  {
104  u_fputc(flexional_traits_separator, f);
105 //The generic u_write function seems not to work for ptr_list<set<unichar> >
106 // u_ptr_write(f, e.possible_flexional_traits.begin(), e.possible_flexional_traits.end(), flexional_traits_separator);
107  u_decompress_delaf_write(f, e.possible_flexional_traits);
108  //Write clitic pronouns, if present
109  if (!e.clitic_pronouns.empty())
110  {
111  u_fputc(clitic_pronouns_separator, f);
112  u_ptr_write(f, e.clitic_pronouns.begin(), e.clitic_pronouns.end(), clitic_pronouns_separator);
