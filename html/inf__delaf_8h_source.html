262  template<typename CharT, typename Traits>
263  std::basic_ostream<CharT, Traits>& operator<< (std::basic_ostream<CharT, Traits> &out, const inf_delaf::entry &e)
264  {
265  if (e.canonical_form_suffix.size())
266  out << e.inflected_form_suffix_length << e.canonical_form_suffix;
267  e.pos.serialize(out << '.');
268 
269  if (!e.semantic_traits.empty())
270  {
271  out << semantic_traits_separator;
272  serialize_ptr_sequence(out, e.semantic_traits.begin(), e.semantic_traits.end(), semantic_traits_separator);
273  }
274 
275  if (!e.possible_flexional_traits.empty())
276  {
277  out << flexional_traits_separator;
278 
279  sequence_serializer<u_set::const_iterator> u_set_serializer;
280 
281  serialize_sequence_of_ptr_sequences(out, e.possible_flexional_traits.begin(), e.possible_flexional_traits.end(), flexional_traits_separator, u_set_serializer);
282  if (!e.clitic_pronouns.empty())
283  {
284  out << clitic_pronouns_separator;
285  serialize_ptr_sequence(out, e.clitic_pronouns.begin(), e.clitic_pronouns.end(), clitic_pronouns_separator);
286  }
287  }
288  return out;
289  }
